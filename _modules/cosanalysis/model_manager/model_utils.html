def check_files(directory_path):
    """
    Check the structure and content of a folder with inference files.
    The following checks are made :
    
    - Ensure the directory exists
    
    - Check if config.json and inference.py files exist
    
    - Check if there is a model file in the directory (the name of the file must be model)
    
    - Check if the JSON file is correct and contains the right arguments
    
    - Check if the run_inference function exists in inference.py
    
    - Check the parameters of the run_inference function in inference.py
    
    - Check if inference.py contains no syntax error
    
    - Check there is no image or csv file in the folder
    
    :param directory_path: Path to the folder with inference files.
    :type directory_path: str

    :return: True if there is no Error raised.
    :rtype: bool

    :raises Error: If a check is not passed.
    """

    # Ensure the directory exists
    if not os.path.isdir(directory_path):
        raise ValueError(
            f"The path {directory_path} is not a valid directory.")

    required_files = {"config.json", "inference.py"}

    # Check if required files exist
    for required_file in required_files:
        file_path = os.path.join(directory_path, required_file)
        if not os.path.isfile(file_path):
            raise ValueError(
                f"Required file {required_file} is missing in the directory."
            )

    # Check if there is a model file in the directory
    model_found = False
    for file in os.listdir(directory_path):
        if os.path.splitext(file)[0] == "model":
            model_found = True
            break
    if not model_found:
        raise ValueError(
            f"No file named 'model' was found in the directory {directory_path}."
        )

    # Check the JSON file
    try:
        config_data = parse_json(directory_path)
        check_json(config_data)

    except json.JSONDecodeError:
        raise ValueError(
            f"{directory_path}/config.json is not a valid JSON file.")

    # Check directory_path/inference.py file
    inference_path = os.path.join(directory_path, "inference.py")
    try:
        with open(inference_path, "r") as inference_file:
            source_code = inference_file.read()
            tree = ast.parse(source_code)

            # Check if the run_inference function exists
            functions = [
                node.name
                for node in ast.walk(tree)
                if isinstance(node, ast.FunctionDef)
            ]
            if "run_inference" not in functions:
                raise ValueError(
                    f"The function 'run_inference' is missing from {directory_path}/inference.py."
                )

            # Check the parameters of the run_inference function
            for node in ast.walk(tree):
                # Look for function definitions
                if isinstance(node, ast.FunctionDef) and node.name == "run_inference":
                    # Extract parameter names
                    actual_parameters = [arg.arg for arg in node.args.args]
                    expected_parameters = [
                        "subjs", "version", "directory_path"]

                    # Add 'savepath' as an optional parameter
                    if "savepath" in actual_parameters:
                        expected_parameters.append("savepath")

                    if actual_parameters != expected_parameters:
                        raise ValueError(
                            f"The function 'run_inference' in {inference_path} does not have the expected parameters. "
                            f"Expected: {expected_parameters}, Found: {actual_parameters}"
                        )
    except SyntaxError:
        raise ValueError(
            f"The file {directory_path}/inference.py contains syntax errors."
        )

    # Check there is no image or csv file
    image_extensions = {".jpg", ".jpeg", ".png",
                        ".bmp", ".gif", ".tiff", ".webp"}
    csv_extension = ".csv"

    for root, dirs, files in os.walk(directory_path):
        for file in files:
            # Extract the extension of the file
            _, extension = os.path.splitext(file)
            # Check if the extension is an image or csv extension
            if (
                extension.lower() in image_extensions
                or extension.lower() == csv_extension
            ):
                raise ValueError(
                    "The directory must not contain any csv or image file."
                )

    return True

def check_used_function(directory_path):
    """
    Check that the `build_analysis_dataset` function is not used in inference.py file
    in the given directory (to ensure the target is not used to predict).

    :param directory_path: Path to the directory containing inference.py.
    :type directory_path: str

    :return: True if `build_analysis_dataset` is not used.
    :rtype: bool

    :raises ValueError: If `build_analysis_dataset` is used.
    """

    inference_path = os.path.join(directory_path, "inference.py")
    with open(inference_path, "r") as inference_file:
        file_content = inference_file.read()

        # Check for the presence of the function name
        if "build_analysis_dataset" in file_content:
            raise ValueError(
                "The function 'build_analysis_dataset' is used in the file inference.py. "
                "It should not be used to run inferences."
            )
    return True

def test_run_inference(directory_path):
    """
    Check that the run_inference function in inference.py runs on a subject of
    the training set of the cohorts specified in the JSON file.
    This function also checks the format of the output of the function, and
    that the timing of the output corresponds to the window_size specified in the JSON file.

    :param directory_path: Path to the directory containing inference.py.
    :type directory_path: str

    :return: True if all checks are passed.
    :rtype: bool

    :raises ValueError: If there is an error during checks.
    """

    config_data = parse_json(directory_path)

    cohort = config_data["training_metadata"]["cohorts"][0]  # first cohort
    version = config_data["training_metadata"]["data_version"]

    # One subject for each cohort
    subj_for_cohort = {
        "PD": "ADQPCTS",
        "IBD": "BAGDKXD",
        "HV": "ACKJXCP",
        "RA": "BCTFCKQ",
        "SLE": "BDKRVYZ",
        "HD": "ACFHJAC",
        "PSS": "BADVEEP",
    }

    subj = [subj_for_cohort[cohort]]

    predictions = run_inference(subj, version, directory_path)

    # Check the format and the columns of the output DataFrame
    check_output(predictions)

    # Check if the timing is right in the prediction
    timing = config_data["dataset_metadata"]["window_size"]
    if timing in ["M", "A", "E", "daily"]:
        unique_timings = predictions.TIMING.unique()
        if not (len(unique_timings) == 1 and unique_timings[0] == timing):
            raise ValueError(
                f"Expected only '{timing}' in TIMING, but found: {unique_timings}"
            )
    else:  # timing=="MAE"
        unique_timings = set(predictions.TIMING.unique())
        required_timings = {"M", "A", "E"}
        if not required_timings.issubset(unique_timings):
            raise ValueError(
                f"Expected predictions to contain 'M', 'A', and 'E' in TIMING, but found: {unique_timings}"
            )

    return True

def run_inference(participants, version, directory_path, savepath=None):
    """
    Run inference on participants using the run_inference function from directory_path/inference.py

    :param participants: List of participants to run inferences on.
    :type participants: list of str

    :param version: Version of the data to use.
    :type version: str

    :param directory_path: Path to the folder with inference files.
    :type directory_path: str

    :return: Predictions for each participants.
    :rtype: pandas.DataFrame

    :raises ValueError: If there is an error while running inferences.
    """
    sys.path.append(directory_path)

    # Run the 'inference.py' file
    inference_path = os.path.join(directory_path, "inference.py")

    spec = importlib.util.spec_from_file_location("inference", inference_path)
    inference_module = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(inference_module)

    # Apply 'run_inference' defined in 'inference.py'
    try:
        predictions = inference_module.run_inference(
            subjs=participants,
            version=version,
            directory_path=directory_path,
            savepath=savepath,
        )
    except Exception as e:
        raise ValueError(f"Error while running inference: {e}")

    return predictions

def fetch_true_target_values(predictions, version=None):
    """
    Fetch the real values corresponding to the timings of the predictions.

    :param predictions: DataFrame with the columns : subject_id, tup, Date, TIMING, prediction
    :type predictions: pandas.DataFrame

    :return: The input DataFrame with the column true_{target_name} corresponding to the real values of the target, and the target name.
    :rtype: tuple of a pandas.DataFrame and a str
    """
    check_output(predictions)

    # Mapping rules for targets to concepts and change_from_baseline
    target_mapping = {
        "overall_fatigue_vas": {"concept": "F", "change_from_baseline": False},
        "mental_fatigue_vas": {"concept": "F", "change_from_baseline": False},
        "physical_fatigue_vas": {"concept": "F", "change_from_baseline": False},
        "sleep_quality_vas": {"concept": "SQ", "change_from_baseline": False},
        "KSS": {"concept": "DS", "change_from_baseline": False},
    }

    # Add "_change" variants
    for target, values in list(target_mapping.items()):
        if not target.endswith("_change"):
            target_mapping[f"{target}_change"] = {
                **values,
                "change_from_baseline": True,
            }

    target_name = predictions.columns[-1].replace("predicted_", "")

    concept = target_mapping[target_name]["concept"]
    change_from_baseline = target_mapping[target_name]["change_from_baseline"]

    if predictions.TIMING.iloc[0] in ["M", "A", "E"]:
        window_size = "intra_daily"
    else:
        window_size = "daily"

    diaries = load_diaries(
        concept=concept,
        window_size=window_size,
        change_from_baseline=change_from_baseline,
        version=version,
        participants=predictions.subject_id.unique(),
    )

    # Filter timings
    diaries = diaries[diaries.TIMING.isin(predictions.TIMING.unique())]

    # Add Date column to merge properly
    diaries["Date"] = pd.to_datetime(diaries.Time).dt.date
    predictions.Date = pd.to_datetime(predictions.Date).dt.date

    diaries.rename(columns={target_name: f"true_{target_name}"}, inplace=True)
    predictions = predictions.merge(
        diaries[["subject_id", "tup", "Date",
                 "TIMING"] + [f"true_{target_name}"]]
    )

    return predictions, target_name

def parse_json(directory_path):
    """
    Parse the config.json file in the directory_path directory.

    config_path = os.path.join(directory_path, "config.json")

        config_data = json.load(config_file)

def check_output(predictions):
    """
    Check the output of the run_inference function in inference.py. The first columns must be ['subject_id', 'tup', 'tup_day', 'Date', 'TIMING'].
    The last column must be the prediction column.

    :param predictions: DataFrame corresponding to the output of the run_inference function of the inference.py file.
    :type predictions: pandas.DataFrame

    :return: True if the output is in the correct format.
    :rtype: bool

    :raises ValueError: If the columns of the DataFrame are not the expected ones.
    """

    if not isinstance(predictions, pd.DataFrame):
        raise TypeError("'predictions' must be a pandas DataFrame.")

    mandatory_columns = ["subject_id", "tup", "tup_day", "Date", "TIMING"]

    # Get the last column and check if it's in the valid list of options
    last_column = predictions.columns[-1]

    valid_last_columns = [
        "predicted_" + col
        for col in [
            "overall_fatigue_vas",
            "physical_fatigue_vas",
            "mental_fatigue_vas",
            "sleep_quality_vas",
            "KSS",
            "overall_fatigue_vas_change",
            "physical_fatigue_vas_change",
            "mental_fatigue_vas_change",
            "sleep_quality_vas_change",
            "KSS_change",
        ]
    ]

    if last_column not in valid_last_columns:
        raise ValueError(
            f"The last column must be one of the following: {', '.join(valid_last_columns)}."
        )

    missing_columns = list(np.setdiff1d(
        mandatory_columns, predictions.columns))
    if len(missing_columns):
        raise ValueError(
            f"The dataset has missing columns: {missing_columns}.")

    # Check that no extra columns are present
    all_valid_columns = mandatory_columns + [last_column]
    if list(predictions.columns) != all_valid_columns:
        raise ValueError(
            f"The DataFrame contains extra columns. Only the following columns are allowed: {', '.join(all_valid_columns)}."
        )

    return True

def check_json(config_data):
    """
    Check if the JSON files contains the required arguments.

    :param config_data: Data of the json file to check.
    :type config_data: dict

    :return: True if the JSON file is correct.
    """
    schema = {
        "properties": {
            "training_metadata": {
                "type": "object",
                "properties": {
                    "cohorts": {
                        "type": "array",
                        "items": {
                            "type": "string",
                            "enum": ["IBD", "HV", "PD", "HD", "PSS", "SLE", "RA"],
                        },
                    },
                    "data_version": {"type": "string"},
                    "supervised": {"type": "boolean"},
                },
                "required": ["cohorts", "data_version", "supervised"],
            },
            "model_metadata": {
                "type": "object",
                "properties": {
                    "library": {"type": "string"},
                    "model_filename": {"type": "string"},
                    "model_type": {"type": "string"},
                    "input_features": {"type": "array", "items": {"type": "string"}},
                },
                "required": [
                    "library",
                    "model_filename",
                    "model_type",
                    "input_features",
                ],
            },
            "dataset_metadata": {
                "type": "object",
                "properties": {
                    "window_size": {
                        "type": "string",
                        "enum": ["MAE", "M", "A", "E", "daily"],
                    },
                    "window_length": {"type": "integer"},
                    "setting": {
                        "type": "string",
                        "enum": ["around", "before", "after"],
                    },
                    "device_types": {
                        "type": "array",
                        "items": {
                            "type": "string",
                            "enum": [
                                "SLP_night",
                                "STS",
                                "COG_pvt",
                                "COG_dst",
                                "MCR_MCR",
                                "MCR_MCL",
                                "HRV",
                                "VIT",
                                "GVA",
                            ],
                        },
                    },
                    "imputation": {"type": "boolean"},
                },
                "required": [
                    "window_size",
                    "device_types",
                    "imputation",
                ],
            },
            "prediction": {
                "type": "object",
                "properties": {
                    "concept": {"type": "string", "enum": ["F", "SQ", "DS"]},
                    "prediction_label": {"type": "string"},
                    "change_from_baseline": {"type": "boolean"},
                    "cohort_specific_model": {"type": "boolean"},
                },
                "required": [
                    "concept",
                    "prediction_label",
                    "change_from_baseline",
                    "cohort_specific_model",
                ],
            },
        },
        "required": [
            "training_metadata",
            "model_metadata",
            "dataset_metadata",
            "prediction",
        ],
    }

    try:
        validate(instance=config_data, schema=schema)
    except ValidationError as e:
        print(f"Validation Error in config.json file : {e}")

    return True
