def upload_model(directory_path):
    """
    Upload model to the DMP. This function also checks the validity of the folder before proceeding to the upload.

    :param directory_path: Path to the folder with the model to upload.
    :type directory_path: str

    :return: True if the upload was a success.
    :rtype: bool

    :note:
        - This function calls check_files, check_used_function and test_run_inference from model_utils to check if the folder is ready to be uploaded. If not, an error will be raised by those functions.
        - The folder is compressed into a zip file. That zip file is the file that will be uploaded to the DMP.

    """
    # Check the folder
    check_files(directory_path)
    check_used_function(directory_path)
    test_run_inference(directory_path)

    # 1- Compress a folder into a zip file
    zip_buffer = BytesIO()
    with zipfile.ZipFile(zip_buffer, "w", zipfile.ZIP_DEFLATED) as zipf:
        for root, dirs, files in os.walk(directory_path):
            for file in files:
                file_path = os.path.join(root, file)
                arcname = os.path.relpath(file_path, directory_path)
                zipf.write(file_path, arcname)
    zip_buffer.seek(0)  # Reset buffer pointer to the beginning

    # 2- Proceed to upload
    # Get model type
    config_data = parse_json(directory_path)

    model_type = config_data["model_metadata"]["model_type"]
    version = config_data["training_metadata"]["data_version"]
    cohorts = config_data["training_metadata"]["cohorts"]
    prediction_window = config_data["dataset_metadata"]["window_size"]
    device_types = config_data["dataset_metadata"]["device_types"]
    prediction_label = config_data["prediction"]["prediction_label"]
    change_from_baseline = config_data["prediction"]["change_from_baseline"]

    file_name = f"model_{model_type}_{prediction_label}_{version}_{int(pd.Timestamp.today().timestamp() * 1000)}.zip"

    result = upload_data(
        study_id=study_id,
        field_id="bi_model",
        file_name=file_name,
        properties={
            "prediction_label": prediction_label,
            "prediction_change_from_baseline": change_from_baseline,
            "prediction_window": prediction_window,
            "model_type": model_type,
            "training_cohorts": cohorts,
            "training_version": version,
            "device_types": device_types
        },
        file_content=zip_buffer,  # Upload the zip buffer
    )
    if result:
        print("The model was uploaded successfully!")

    return result

def run_inference_on_analysis_set(
    model_id,
    version,
    model_path="./models",
    cohorts=None,
    analysis_set="ias",
    ias_dataset="Train",
    savepath=None,
):
    """
    Run inference on a chosen analysis set. For cohort specific models, we will only be able to test on participants of the specific cohort.

    :param model_id: ID of the model on the DMP.
    :type model_id: str

    :param version: Version of the data to use.
    :type version: str

    :param model_path: Path to save the model folder to. Default is "./models" : model is stored in the folder models at the current path.
    :type model_path: str, optional

    :param cohorts: Cohorts to run inferences on. Default is None : we test on all cohorts (except if this is a cohort specific model).
    :type cohorts: list of str

    :param analysis_set: Analysis set to run inferences on. Either 'ias' or 'eas'. Default is 'ias'
    :type analysis_set: str, optional

    :param ias_dataset: Case where analysis_set = 'ias'. Indicates whether to run inferences on train set, test set or both. Must be one of 'Train', 'Test', 'all'. Default is 'Train'.
    :type ias_dataset: str, optional

    :param savepath: Path to save the predictions to. Default is None : the predictions dataset will not be saved.
    :type savepath: str, optional

    :return: DataFrame with the predictions corresponding to the analysis set in parameter.
    :rtype: pandas.DataFrame

    :note: This function raises ValueError in case analysis_set or ias_dataset are not valid.
    """
    models_info = pd.DataFrame(list_files(
        study_id, kinds=["bi_model"], version_id=None))

    if models_info[models_info.fileId == model_id].empty:
        raise ValueError(f"Model {model_id} not found.")

    file_info = models_info[models_info.fileId == model_id].iloc[0]

    file_name = file_info.fileName[:-4]

    record_stream = get_file_content(model_id)
    compressed_data = BytesIO(record_stream)

    with zipfile.ZipFile(compressed_data) as zf:
        with zf.open("config.json") as file:
            config_data = json.load(file)

        # Define model saving path
        os.makedirs(model_path, exist_ok=True)
        extract_path = os.path.join(model_path, file_name)
        zf.extractall(extract_path)
            
    if config_data["prediction"]["cohort_specific_model"]:  # Bool
        allowed_cohorts = config_data["training_metadata"]["cohorts"]

        if cohorts is None:
            cohorts = allowed_cohorts
        else:
            cohorts = [
                cohort for cohort in cohorts if cohort in allowed_cohorts]

    # Get the subjects from the identification set
    clinical_df = read_adam_dataset(data_type="ADSL",
                                    version=version,
                                    imputation=False,
                                    analysis_set=analysis_set,
                                    ias_dataset=ias_dataset,
                                    cohorts=cohorts)

    subjects = clinical_df.subject_id.unique()

    predictions = run_inference(subjects, version, extract_path)

    if savepath is not None:
        predictions.to_csv(savepath, index=False)

    return predictions
