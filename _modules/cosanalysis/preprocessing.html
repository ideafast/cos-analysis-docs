def processing(dataset, normalize="z_score"):
    """
    Preprocess the given dataset by normalizing the data, encoding categorical variables,
    and removing columns with no variability.

    :param dataset: The input dataframe to be processed. It should contain the device data,
                    diary data, and possibly other columns for analysis.
    :type dataset: pandas.DataFrame

    :param normalize: Specifies the normalization method to apply to the numerical columns:
                      - 'z_score': Normalize using z-score (standardization).
                      - 'min_max': Normalize using Min-Max scaling.
                      - None: No normalization applied.
                      Default is 'z_score'.
    :type normalize: {'z_score', 'min_max', None}, optional

    :return: A processed dataframe with the specified normalization applied, categorical variables
              encoded, and columns with no variability removed.
    :rtype: pandas.DataFrame

    :raises ValueError: If the `normalize` parameter is not one of ['z_score', 'min_max', None].

    :note:
        - Non-numeric columns such as `subject_id`, `tup`, `tup_day`, `start_window` and `stop_window` are excluded from normalization.
        - Categorical columns are mapped to numeric values (e.g., 'GENDER', 'COHORT') using a predefined
          mapping in the `label_mappings` function.
        - One-hot encoding is performed on columns starting with certain prefixes, such as
          `VIT_PostureCategory_1min`, `mcr_class`, and `mcl_class`.
        - Any columns with constant values (no variability) are dropped before returning the result.
    """

    result_df = dataset.copy()
    # Check the normalize parameter
    if normalize not in ["z_score", "min_max", None]:
        raise ValueError(
            'The parameter "normalize" must be one of ["z_score", "min_max", None]'
        )

    # Label the string columns
    # 1 - diaries
    mappings = label_mappings()
    for col, _ in mappings.items():
        if col in result_df.columns:
            result_df[col] = result_df[col].map(mappings[col])

    # 2 - device (One Hot Encoding)
    one_hot_columns = ["VIT_PostureCategory_1min", "mcr_class", "mcl_class"]
    to_encode = [
        col
        for col in result_df.columns
        if any(col.startswith(prefix) for prefix in one_hot_columns)
    ]

    encoded_col = []
    if to_encode != []:
        result_df = pd.get_dummies(result_df, columns=to_encode)
        encoded_col = [col for col in result_df.columns if col not in dataset]

    # Convert numeric columns
    not_numeric = ["subject_id", "tup", "tup_day", "start_window", "stop_window"]
    cols_to_convert = [col for col in result_df.columns if col not in not_numeric]
    result_df[cols_to_convert] = result_df[cols_to_convert].apply(pd.to_numeric)

    # Drop columns with no variability
    cols_to_drop = [col for col in result_df.columns if result_df[col].nunique() <= 1]
    result_df = result_df.drop(columns=cols_to_drop)

    # Normalize if requested
    if normalize is not None:
        cols_to_normalize = [
            col
            for col in result_df.columns
            if col not in not_numeric
            and col not in encoded_col
            and col not in list(mappings.keys())
        ]
        # we replace NaN values by 0 temporarily - before the imputation is complete
        # Save the positions of NaN values
        nan_mask = result_df[cols_to_normalize].isna()
        # Fill NaNs with 0 temporarily for normalization
        result_df[cols_to_normalize] = result_df[cols_to_normalize].fillna(0)

        if normalize == "z_score":
            result_df[cols_to_normalize] = result_df[cols_to_normalize].apply(zscore)
        else:
            # We will have to save that scaler
            scaler = MinMaxScaler()
            result_df[cols_to_normalize] = scaler.fit_transform(
                result_df[cols_to_normalize]
            )

        # Restore NaN values to their original positions
        result_df[cols_to_normalize] = result_df[cols_to_normalize].mask(nan_mask)

    return result_df.copy()

def label_mappings():
    """
    Returns a dictionary of label mappings for various columns, such as `TIMING`, `GENDER`, `COHORT`,
    `Season`, and `Day_Type`.

    :return: A dictionary containing the mappings for each column.
    :rtype: dict
    """
    return {
        "TIMING": {
            "M": 1,
            "A": 2,
            "E": 3,
            "daily": 4,
        },
        "GENDER": {"Male": 0, "Female": 1},
        "COHORT": {
            "HV": 0,
            "PD": 1,
            "HD": 2,
            "PSS": 3,
            "IBD": 4,
            "SLE": 5,
            "RA": 6,
            "": 7,  # Empty or Unknown
        },
        "Season": {"Winter": 1, "Spring": 2, "Summer": 3, "Fall": 4},
        "Day_Type": {"Weekday": 0, "Weekend": 1},
    }
