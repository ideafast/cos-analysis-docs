def upload_dataset(
    dataset,
    data_type,
    version=None,
    imputation=False,
    analysis_set="ias"
):
    """
    Upload a dataset on the DMP, with the field id {data_type}_agg.

    :param dataset: Dataset to upload.
    :type dataset: pandas.DataFramr

    :param data_type: The type of the data we want to upload. 
        Must be one of :
        ["ADDI", "ADSL", "ADCL", "ADPRO", "SLP_night", "STS", "HRV", 
        "VIT", "GVA", "MCR_MCR", "MCR_MCL", "COG_pvt", "COG_dst"]
    :type data_type: str

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: If True, we load the imputed dataset. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" (identification) or "eas" (evaluation). Default is "ias".
    :type analysis_set: str, optional

    :return: The result of the upload process.
    """
    # Save on the DMP
    data_type = check_valid_type(data_type)

    field_id = f"{data_type}_agg"

    version_id = get_version_id(version)

    if imputation:
        file_name = f"{data_type}_imputed_dataset_{version_id}_{pd.Timestamp.today().date()}.csv"
    else:
        file_name = (
            f"{data_type}_dataset_{version_id}_{pd.Timestamp.today().date()}.csv"
        )

    csv_buffer = StringIO()
    dataset.to_csv(csv_buffer, index=False)
    csv_buffer.seek(0)

    result = upload_data(
        study_id=study_id,
        field_id=field_id,
        file_name=file_name,
        properties={
            "version": version_id,
            "version_tag": version,
            "imputation": str(imputation),
            "analysis_set": analysis_set,
        },
        file_content=csv_buffer,
    )
    return result

def read_dataset(data_type, version=None, imputation=False, analysis_set="ias"):
    """
    Read a dataset that is saved on the DMP, chosing the data_type, the version, if the dataset is imputed and the analysis set.

    :param data_type: The type of the data we want to load. 
        Must be one of :
        ["ADDI", "ADSL", "ADCL", "ADPRO", "SLP_night", "STS", "HRV", 
        "VIT", "GVA", "MCR_MCR", "MCR_MCL", "COG_pvt", "COG_dst"]
    :type data_type: str

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: If True, we load the imputed dataset. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" (identification) or "eas" (evaluation). Default is "ias".
    :type analysis_set: str, optional

    :return: The pre-saved dataset with the required properties.
    :rtype: pandas.DataFrame

    :raises ValueError: If no file is found (probably a permission issue) or if no file_id corresponds to the version, imputation and analysis_set given.
    """
    data_type = check_valid_type(data_type)
    analysis_set = check_valid_dataset(analysis_set)

    field_id = f"{data_type}_agg"

    files = list_files(study_id=study_id, kinds=[field_id], version_id=None)

    files_df = pd.DataFrame(files)

    if files_df.empty:
        raise ValueError("The files DataFrame is empty. Please check your data access permissions.")
        
    if version is None:
        fileids = files_df[
            files_df["properties"].apply(
                lambda x: x.get("imputation") == str(imputation)
                and x.get("analysis_set") == analysis_set
            )
        ]
    else:
        fileids = files_df[
            files_df["properties"].apply(
                lambda x: x.get("version_tag") == version
                and x.get("imputation") == str(imputation)
                and x.get("analysis_set") == analysis_set
            )
        ]

    if fileids.empty:
        raise ValueError(
            f"No saved file correspond to the field_id {field_id}, "
            f"the analysis set {analysis_set}, "
            f"the version {version} and the imputation {imputation}."
        )

    fileid = fileids.sort_values("uploadTime", ascending=False).iloc[0]

    file_stream = get_file_content(fileid.fileId)
    dataset = pd.read_csv(BytesIO(file_stream))

    return dataset.copy()

def get_tup_dates(version=None, analysis_set="ias"):
    """
    Get the tup start and stop dates.

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :return: A DataFrame with the start and stop dates for each TUP of each subject.
    :rtype: pandas.DataFrame
    """

    # read participant level information
    participants = read_dataset(
        data_type="ADSL", version=version, imputation=False, analysis_set=analysis_set)

    participants = participants.rename(columns={"USUBJID": "subject_id"})

    tup_dates = participants.sort_values(by="subject_id").melt(
        id_vars=["subject_id"],
        value_vars=["TUP1DT", "TUP2DT", "TUP3DT", "TUP4DT"],
        var_name="TUP",
        value_name="start_date",
    )

    tup_dates = tup_dates.replace("", np.nan).dropna()
    tup_dates.TUP = tup_dates.TUP.map(
        {"TUP1DT": 1, "TUP2DT": 2, "TUP3DT": 3, "TUP4DT": 4}
    )
    tup_dates.rename(
        columns={
            "TUP": "tup",
        },
        inplace=True,
    )

    tup_dates["start_date"] = pd.to_datetime(tup_dates.start_date)
    tup_dates["stop_date"] = tup_dates.start_date + timedelta(days=7)
    tup_dates.loc[tup_dates.tup == 1, "stop_date"] = tup_dates.start_date + timedelta(
        days=10
    )
    tup_dates = tup_dates.sort_values(["subject_id", "tup"])
    tup_dates = tup_dates.drop_duplicates(
        subset=["subject_id", "start_date"], keep="first"
    ).reset_index(drop=True)

    return tup_dates

def add_tups(
    data,
    tup_dates,
    lag_days_before=3,
    lag_days_after=4,
    tup_header="tup",
    subject_header="subject_id",
    time_header="Time",
):
    """
    Add a column TUP to the input vtp_records dataframe with
    the corresponding TUP.

    :param data: input dataframe where TUP mapping is applied
    :type data: pandas.DataFrame

    :param lag_days_before: nb of allowed days before TUP starts
    :type lag_days_before: int

    :param lag_days_after: nb of allowed days after TUP ends
    :type lag_days_after: int

    :param tup_header: Name of TUP column to be added in input data
    :type tup_header: str

    :param subject_header: Participant header in input data
    :type subject_header: str

    :param time_header: Time header in input data
    :type time_header: str

    :return: Input Dataframe with column TUP
    :rtype: pandas.DataFrame
    """
    tup_dates = tup_dates.copy()
    tup_dates = tup_dates[tup_dates.subject_id.isin(
        data[subject_header].unique())]
    tup_dates.rename(columns={"subject_id": subject_header}, inplace=True)
    data[time_header] = pd.to_datetime(data[time_header])

    merged_data = pd.merge(
        data, tup_dates, on=["subject_id"], how="left", suffixes=(None, "_y")
    )
    tup_dates_header = "tup_y" if "tup_y" in merged_data.columns else "tup"
    start_header = (
        "start_date_y" if "start_date_y" in merged_data.columns else "start_date"
    )
    stop_header = "stop_date_y" if "stop_date_y" in merged_data.columns else "stop_date"

    # mapping between input dataset and tup dates
    merged_data = (
        merged_data[
            (
                merged_data[time_header]
                >= merged_data[start_header] - timedelta(days=lag_days_before)
            )
            & (
                merged_data[time_header]
                <= merged_data[stop_header] + timedelta(days=lag_days_after)
            )
        ]
        .reset_index(drop=True)
        .copy()
    )

    # if 2 tups almost intercept - keep the closest one
    merged_data["in_tup"] = (merged_data[time_header] >= merged_data[start_header]) & (
        merged_data[time_header] <= merged_data[stop_header]
    )
    merged_data = merged_data.sort_values(
        by=[subject_header, time_header, "in_tup", start_header],
        ascending=[True, True, False, True],
    )
    merged_data = merged_data.drop_duplicates(list(data.columns), keep="first")

    # add tup column to input dataset
    data_with_tup = data.merge(merged_data, how="left")
    data_with_tup[tup_header] = data_with_tup[tup_dates_header]
    data_with_tup.fillna(value={tup_header: -1}, inplace=True)
    data_with_tup.drop(
        columns=np.setdiff1d(
            list(data_with_tup.columns), list(data.columns) + [tup_header]
        ),
        inplace=True,
    )
    return data_with_tup

def add_tup_day(
    data,
    tup_dates,
    tup_header="tup",
    subject_header="subject_id",
    time_header="Time",
):
    """
    Add a column with the day number within the TUP.

    :param data: Input DataFrame to add the tup_day column to.
    :type data: pandas.DataFrame

    :param tup_dates: DataFrame with the start and stop dates of each patients tups.
    :type tup_dates: pandas.DataFrame

    :param tup_header:  Name of the TUP column. Default is "tup".
    :type tup_dates: str, optional

    :param subject_header: Name of the subject ID column. Default is "subject_id".
    :type subject_header: str, optional

    :param time_header: Name of the time column. Default is "Time".
    :type time_header: str, optional

    :return: Input DataFrame with column tup_day
    :rtype: pandas.DataFrame
    """

    data = data.copy()

    data = data.merge(tup_dates, on=[subject_header, tup_header], how="left")

    data["tup_day"] = (
        pd.to_datetime(data[time_header]) - pd.to_datetime(data.start_date)
    ).dt.days

    data = data.drop(columns=["start_date", "stop_date"])
    return data.copy()

def get_version_id(version=None):
    """
    Get version ID corresponding to a version tag.

    :param version: Version tag corresponding to the version_id we want to find. Default is None : the version_id returned is also None.
    :type version: str, optional

    :return: The corresponding version_id if version is not None.
    :rtype: str, optional

    :raises ValueError: If the version does not exist.
    """
    if version is None:
        return None
    else:
        study_versions = list_study_versions(study_id)
        version_id = study_versions[study_versions.tag == version]
        if version_id.empty:
            raise ValueError(
                f"Invalid version: {version}. Must be one of {study_versions.tag.unique()}."
            )
        version_id = str(version_id.version.values[0])
    return version_id

def check_cohorts(cohorts):
    """
    Check if the cohorts are valid. Each must be one of ["HV", "PD", "HD", "IBD", "SLE", "PSS", "RA"].

    :param cohorts: Cohorts to check.
    :type cohorts: List of str

    :return: The cohorts list.

    :raises ValueError: If cohorts is not a list of string or if there is an invalid cohort in the list.
    """
    valid_cohorts = ["HV", "PD", "HD", "IBD", "SLE", "PSS", "RA"]

    if cohorts is None:
        cohorts = valid_cohorts

    if isinstance(cohorts, str):
        cohorts = [cohorts]

    if not isinstance(cohorts, list):
        raise ValueError("'cohorts' must be a list of string.")

    if not all([c in valid_cohorts for c in cohorts]):
        raise ValueError(
            f"Invalid 'cohorts'. Cohorts must be in {valid_cohorts}.")

    return cohorts

def get_device_measures(data):
    """
    Retrieves the device measures from a given dataset.

    :param data: The input DataFrame containing device data, where the column names represent various device measures.
    :type data: pandas.DataFrame

    :return: A list of column names from the input DataFrame that correspond to device measures.
    :rtype: list

    :note: The matching is case-insensitive and uses the starting substring match.
    """

    valid_device_types = get_valid_devices()
    measures_code = [dd for d in valid_device_types for dd in d.split("_")]
    measures = [
        c
        for c in data.columns
        if c.lower().startswith(tuple([c.lower() for c in measures_code]))
    ]
    return measures

def check_valid_dataset(dataset):
    """
    Check if dataset is either 'ias' or 'eas'.

    :param dataset: Parameter to check. Must be either 'ias' or 'eas'.
    :type dataset: str

    :return: dataset if no error was raised.
    :rtype: str

    :raises ValueError: If dataset is neither 'ias' nor 'eas' or (temporary) if dataset is 'eas' (forbidden access).
    """
    valid_datasets = ["ias", "eas"]
    dataset = dataset.lower()
    if dataset == "isa":
        dataset = "ias"

    if dataset not in valid_datasets:
        raise ValueError(
            f"Invalid 'dataset'. Dataset must be in {valid_datasets}.")
    elif dataset == "eas":
        raise ValueError(f"Forbidden access")
    return dataset

def check_participants(participants):
    """
    Check if the participants is in the right format.

    :param participants: Parameter to check.
    :type participants: str, list of str or None.

    :return: None if participants is None, and a list of participants if the format is correct.
    :rtype: list of str or None

    :raises ValueError: If participants is not a string, a list of string or None.
    """
    if participants is None:
        return None
    if isinstance(participants, str):
        return [participants]
    if isinstance(participants, np.ndarray):
        participants = participants.tolist()
    if isinstance(participants, list) and all(
        isinstance(item, str) for item in participants
    ):
        return participants
    raise ValueError(
        "Participants must be a either a string, a list of string or None."
    )
