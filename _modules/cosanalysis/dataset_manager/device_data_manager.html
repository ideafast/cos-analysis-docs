def read_device_dataset(
    device_type,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    cohorts=None,
):
    """
    Reads a pre-saved dataset, filtering by cohorts, participants and analysis set. The data of each participant is cleaned and concatenated.

    :param device_type: The type of device. Must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR', 'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_type: str

    :param version: Specifies the version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: The analysis set to use ('ias' or 'eas'). Default is 'ias'.
    :type analysis_set: str, optional

    :param participants: A list of participant IDs to include in the dataset. Default is None, all participants for the dataset (ias or eas) are loaded.
    :type participants: list of str, optional

    :param ias_dataset: Specifies the dataset of the identification set to load. Must be one of 'Train', 'Test', or 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: A list of cohorts to load. Default is None, all cohorts are loaded.
    :type cohorts: list of str, optional

    :return: A DataFrame containing the filtered dataset.
    :rtype: pandas.DataFrame

    :raises ValueError: If no file corresponding to the given parameters is found.

    """
    cohorts = check_cohorts(cohorts)
    device_type = check_valid_device(device_type)
    analysis_set = check_valid_dataset(analysis_set)
    ias_dataset = check_valid_ias_dataset(ias_dataset)
    participants = check_participants(participants)

    device_dataset = read_dataset(
        device_type, version, imputation, analysis_set)

    clinical_df = read_adam_dataset(data_type="ADSL",
                                    version=version,
                                    imputation=False,
                                    analysis_set=analysis_set,
                                    participants=participants,
                                    ias_dataset=ias_dataset,
                                    cohorts=cohorts)
    subjects = clinical_df.subject_id.unique()

    device_dataset = device_dataset[device_dataset.subject_id.isin(subjects)]

    device_dataset.start_timestamp = pd.to_datetime(
        device_dataset.start_timestamp, format="mixed"
    )
    device_dataset.end_timestamp = pd.to_datetime(
        device_dataset.end_timestamp, format="mixed"
    )

    return device_dataset.copy()

def load_and_save_device_dataset(device_type, version=None, upload=False, analysis_set="ias"):
    """
    Reads, concatenates, cleans, and optionally uploads a device dataset.

    :param device_type: The type of device. Must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR', 'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_type: str

    :param version: Specifies the version of the data to use. Default is None.
    :type version: str, optional

    :param upload: Whether to upload the cleaned dataset. Default is False.
    :type upload: bool, optional

    :param analysis_set: The analysis set to use ('ias' or 'eas'). Default is 'ias'.
    :type analysis_set: str, optional

    :return: A DataFrame containing the cleaned dataset, or None if no data is available after cleaning.
    :rtype: pandas.DataFrame
    """

    device_type = check_valid_device(device_type)
    analysis_set = check_valid_dataset(analysis_set)

    if "_" in device_type:
        device, level = device_type.split("_")
    else:
        device = device_type

    # Read and concatenate device data
    records = read_device_records(device=device,
                                  version=version,
                                  analysis_set=analysis_set)

    # Get the start and end dates of each patients tups
    tup_dates = get_tup_dates(version, analysis_set)

    # For AX6 device, keep only device worn on the back
    if device_type in ["MCR_MCR", "MCR_MCL", "STS"]:
        ax6_records = list_files(study_id, kinds=["AX6"])
        ax6_records = pd.DataFrame(ax6_records)

        ax6_back = ax6_records[
            ax6_records["properties"].apply(
                lambda x: x.get("location") == "back")
        ]
        records_back = ax6_back.fileName.unique()

        records["ax6_filename"] = records.fileName.str.replace(
            f"-{device}", "-AX6")
        records = records[records.ax6_filename.isin(records_back)].copy()

    # Create a progress bar for the main process
    pbar = tqdm(total=len(records), desc=f"Processing {device_type} records")
    num_processes = 8

    def update_pbar(_):
        # Callback function to update progress bar
        pbar.update(1)

    with mp.Pool(num_processes) as pool:
        # Use imap_unordered for potentially faster processing
        # Apply_async with callback for immediate progress updates
        results = [
            pool.apply_async(
                read_record_wrapper,
                args=(record, device_type, tup_dates),
                callback=update_pbar,
            )
            for record in records.to_dict("records")
        ]
        # Get all results
        all_data = [res.get() for res in results]

    pbar.close()

    all_data = [data for data in all_data if data is not None]
    if all_data:
        device_dataset = pd.concat(all_data, ignore_index=True)
    else:
        print("No data to concatenate.")
        return None

    # Convert timestamps to datetime
    device_dataset.start_timestamp = pd.to_datetime(
        device_dataset.start_timestamp)
    device_dataset.end_timestamp = pd.to_datetime(device_dataset.end_timestamp)

    # device measures
    measures = [
        col
        for col in device_dataset.columns
        if col
        not in ["subject_id", "tup", "device_id", "start_timestamp", "end_timestamp"]
    ]

    # Ensure all datasets have a uniform column structure
    if device_type == "GVA":
        device_dataset.tup = device_dataset.tup.replace(
            "Unscheduled Visit", np.nan)

        # Remove rows with no data
        device_dataset = device_dataset[device_dataset.gva_all_NumSteps_TOTAL != 0]

        # Add the gva_ suffix to the length_recording_hours column
        device_dataset.rename(
            columns={"length_recording_hours": "gva_length_recording_hours"},
            inplace=True,
        )

    elif device_type in ["COG_dst", "COG_pvt"]:
        # Ensure the cognition dst data falls within the evening hours
        if device_type == "COG_dst":
            device_dataset = device_dataset[
                (device_dataset.start_timestamp.dt.hour >= 17)
                & (device_dataset.start_timestamp.dt.hour <= 23)
            ]
        # Ensure the cognition pvt data falls within the morning hours
        else:
            device_dataset = device_dataset[
                (
                    (device_dataset.start_timestamp.dt.hour >= 6)
                    & (device_dataset.start_timestamp.dt.hour <= 12)
                )
            ]

        # Drop duplicates
        device_dataset["Date"] = device_dataset.start_timestamp.dt.date
        device_dataset.drop_duplicates(
            subset=["subject_id", "Date"], keep="first", inplace=True
        )
        device_dataset.drop(columns=["Date"], inplace=True)

    elif device_type == "STS":
        # Drop duplicates
        device_dataset.drop_duplicates(
            subset=["subject_id", "tup", "start_timestamp"], keep="first", inplace=True
        )

    elif device_type in ["MCR_MCR", "MCR_MCL"]:
        # Replace NA values by nan
        if device_type == "MCR_MCL":
            device_dataset["mcl_size"] = pd.to_numeric(
                device_dataset["mcl_size"].replace("NA", np.nan)
            )
            device_dataset["mcl_mi"] = pd.to_numeric(
                device_dataset["mcl_mi"].replace("NA", np.nan)
            )
        # Filter when the device is not worn
        else:
            device_dataset = device_dataset[device_dataset.mcr_class != "not_worn"]

    elif device_type == "HRV":
        device_dataset.rename(columns={"HR_5min": "HRV_HR_5min"}, inplace=True)

    elif device_type == "VIT":
        # Remove rows where only RRI is not nan
        device_dataset.dropna(
            subset=[m for m in measures if "RRI" not in m], how="all", inplace=True
        )
    elif device_type == "SLP_night":
        # remove short records <= 3h
        device_dataset = device_dataset[device_dataset.end_timestamp -
                                        device_dataset.start_timestamp >= datetime.timedelta(hours=3)]
        # remove records with invalid dates
        device_dataset = device_dataset[(device_dataset.start_timestamp.dt.hour < 10) | (
            device_dataset.start_timestamp.dt.hour >= 18)]
        device_dataset = device_dataset[device_dataset.end_timestamp.dt.hour < 18].copy(
        )
        # remove duplicates
        device_dataset["date"] = device_dataset.end_timestamp.dt.date - \
            datetime.timedelta(days=1)
        device_dataset.drop_duplicates(
            subset=["subject_id", "tup", "date", "start_timestamp", "end_timestamp"], inplace=True)

        # remove outliers
        device_dataset.loc[device_dataset.slp_HeartrateQ05_night <
                           30, "slp_HeartrateQ05_night"] = np.nan
        device_dataset.loc[device_dataset.slp_HeartrateQ50_night <
                           30, "slp_HeartrateQ50_night"] = np.nan
        device_dataset.loc[device_dataset.slp_HeartrateQ95_night <
                           30, "slp_HeartrateQ95_night"] = np.nan
        device_dataset.loc[device_dataset.slp_ResprateQ05_night <
                           3, "slp_ResprateQ05_night"] = np.nan
        device_dataset.loc[device_dataset.slp_ResprateQ50_night <
                           3, "slp_ResprateQ50_night"] = np.nan
        device_dataset.loc[device_dataset.slp_ResprateQ95_night <
                           3, "slp_ResprateQ95_night"] = np.nan
        device_dataset.reset_index(drop=True, inplace=True)
        device_dataset.drop(
            columns=["slp_SleepEfficiencyOrigin_night", "date"], inplace=True)

    # Remove NaN columns
    device_dataset.dropna(axis=1, how="all", inplace=True)

    # Replace infinite values by NaN
    device_dataset.replace([np.inf, -np.inf], np.nan, inplace=True)

    # Remove NaNs rows
    measures = [m for m in measures if m in device_dataset.columns]
    device_dataset.dropna(subset=measures, how="all", inplace=True)

    # Sort the dataset
    device_dataset.reset_index(drop=True, inplace=True)
    device_dataset.sort_values(
        by=["subject_id", "tup", "start_timestamp"], inplace=True
    )

    # Drop duplicates
    # device_dataset.drop_duplicates(keep="first", inplace=True)
    device_dataset.drop_duplicates(
        subset=["subject_id", "tup", "device_id",
                "start_timestamp", "end_timestamp"],
        keep="first",
        inplace=True,
    )

    # Add tup day
    if device_type == "SLP_night":
        device_dataset = add_tup_day(
            device_dataset,
            tup_dates,
            time_header="end_timestamp",
        )
        device_dataset.tup_day = device_dataset.tup_day - 1
    else:
        device_dataset = add_tup_day(
            device_dataset,
            tup_dates,
            time_header="start_timestamp",
        )

    # Order columns
    first_columns = [
        "subject_id",
        "device_id",
        "tup",
        "tup_day",
        "start_timestamp",
        "end_timestamp",
    ]
    order = first_columns + [
        col for col in device_dataset.columns if col not in first_columns
    ]
    device_dataset = device_dataset[order]

    if upload:
        result = upload_dataset(device_dataset,
                                device_type,
                                version,
                                imputation=False,
                                analysis_set=analysis_set
                                )

    return device_dataset.copy()

def read_record_wrapper(row_dict, device_type, tup_dates):
    """
    A wrapper function to handle dictionary input for multiprocessing.

    :param row_dict: A dictionary containing record information.
    :type row_dict: dict

    :param device_type: The type of device (e.g., 'SLP_night', 'STS', etc.).
    :type device_type: str

    :param tup_dates: DataFrame containing Tup dates for the records.
    :type tup_dates: pd.DataFrame

    :return: Data read from the archive matching the specified parameters.
    :rtype: pd.DataFrame
    """
    return read_record(pd.Series(row_dict), device_type, tup_dates)

def read_device_records(device, version, analysis_set):
    """
    Retrieves records with the digital measures of a specific device.

    :param device: The type of device. Must be one of 'SLP', 'STS', 'HRV', 'VIT', 'GVA', 'MCR', 'COG'.
    :type device: str

    :param version: The version of the data to retrieve.
    :type version: str

    :param analysis_set: The analysis set to use ('ias' or 'eas')
    :type analysis_set: str

    :return: A DataFrame containing the device records.
    :rtype: pandas.DataFrame
    """

    clinical_df = read_adam_dataset(data_type="ADSL",
                                    version=version,
                                    imputation=False,
                                    analysis_set=analysis_set,
                                    ias_dataset="all")

    participants = clinical_df.subject_id.unique()

    files = list_files(
        study_id,
        kinds=[device],
        participants=participants,
        version_id=version,
    )

    records = pd.DataFrame(files).rename(columns={"participantId": "USUBJID"})

    records.timeStart = pd.to_datetime(records.timeStart)
    records.timeEnd = pd.to_datetime(records.timeEnd)
    records.fileSize = records.fileSize.astype(int)
    records["log_fileSize"] = np.log10(records["fileSize"]).round()

    records = records.sort_values(
        ["USUBJID", "timeStart", "fileSize"]).reset_index(drop=True)

    # remove duplicated records (tolerance of 2 days in timestart)
    records["group_id_start"] = records.groupby(["USUBJID"])["timeStart"]\
        .apply(lambda x: (x.diff().dt.days > 2).cumsum())\
        .reset_index(drop=True)
    records["group_id"] = records.groupby(["USUBJID"])[["timeStart", "timeEnd"]]\
        .apply(lambda x: (
            (x["timeStart"].diff().abs().dt.days > 2)
            | (x["timeEnd"].diff().abs().dt.days > 2)
        ).cumsum())\
        .reset_index(drop=True)

    records = records.sort_values(
        ["USUBJID", "group_id", "timeUpload", "fileSize"]).reset_index(drop=True)
    records.drop_duplicates(
        subset=["USUBJID", "deviceId", "group_id", "log_fileSize"], keep="last", inplace=True)

    records = records.sort_values(
        ["USUBJID", "deviceId", "group_id_start", "fileSize"]).reset_index(drop=True)
    records.drop_duplicates(
        subset=["USUBJID", "deviceId", "group_id_start"], keep="last", inplace=True)

    return records

def read_record(row, device_type, tup_dates):
    """
    Reads a specific device record based on the provided parameters.

    :param row: A row containing the record information.
    :type row: pandas Series

    :param device_type: The type of device. Must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR', 'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_type: str

    :param tup_dates: DataFrame containing Tup dates for the records.
    :type tup_dates: pandas.DataFrame

    :return: DataFrame containing the read data from the archive matching the specified parameters.
    :rtype: pandas.DataFrame

    :raises Exception: If there is an error while reading the data.
    """
    record_id = row["fileId"]
    record_name = row["fileName"]

    if "_" not in device_type:
        record = os.path.splitext(record_name)[0]
        file_regex = f"^{record}.csv$"
    elif device_type in ["MCR_MCR", "MCR_MCL"]:
        level = device_type.split("_")[-1]
        file_regex = rf"{level}"
    else:
        level = device_type.split("_")[-1]
        file_regex = f".+_{level}.csv$"

    try:
        data = read_data_from_archive(record_id, file_regex)

    except Exception as e:
        print(f"Failed to read data for {record_id} - {record_name}: {str(e)}")
        return None

    if data.empty:
        return None

    # to datetime
    data.start_timestamp = pd.to_datetime(data.start_timestamp, format="mixed")
    data.end_timestamp = pd.to_datetime(data.end_timestamp, format="mixed")

    # Map tup
    data = add_tups(data, tup_dates, time_header="end_timestamp")
    data = data.loc[data["tup"] != -1]

    return data

def get_AX_location(file_id):
    """
    Determine the device location (wrist or back).

    :param file_id: The unique identifier of the file to process.
    :type file_id: str

    :return: A string indicating the device location ("wrist" or "back").
             Returns "unknown" if the location cannot be determined or an error occurs.
    :rtype: str

    :raises Exception: If the file cannot be read or processed, the exception is caught and logged.
    """

    try:
        data = read_data_from_archive(
            file_id, "^measurement_.*_classification_24h.csv$", sep=";"
        )
        visit = data.visit_name.unique()[0]
    except Exception as e:
        print(f"Failed to read data for {file_id}: {str(e)}")
        return "unknown"

    location = "wrist" if "wrist" in visit else "back"
    return location

def read_data_from_archive(record_id, file_regex, sep=","):
    """
    Extract and read a specific file from a compressed archive.

    This function searches for a file within a ZIP archive that matches a given regex pattern,
    reads its content, and returns it as a DataFrame.

    :param record_id: The unique identifier of the archive to process.
    :type record_id: str

    :param file_regex: The regular expression pattern to match the desired file inside the archive.
    :type file_regex: str

    :param sep: The delimiter used in the file to separate values (default is ",").
    :type sep: str, optional

    :return: A DataFrame containing the data from the matched file.
    :rtype: pandas.DataFrame

    :raises FileNotFoundError: If no file matching the regex is found in the archive.
    :raises UnicodeDecodeError: If the file content cannot be decoded using UTF-8.
    """

    record_stream = get_file_content(record_id)
    compressed_data = BytesIO(record_stream)

    df = None
    with zipfile.ZipFile(compressed_data) as zf:
        for file_info in zf.infolist():
            if not re.search(file_regex, file_info.filename):
                continue

            with zf.open(file_info, "r") as file:
                try:
                    data = StringIO(file.read().decode("utf-8")).getvalue()
                    df = pd.read_csv(StringIO(data), sep=sep)
                    break
                except UnicodeDecodeError:
                    raise UnicodeDecodeError(
                        f"Could not decode file {file_info.filename} in UTF-8"
                    )

    if df is None:
        raise FileNotFoundError(
            f"Could not find file matching regex '{file_regex}'.")

    return df
