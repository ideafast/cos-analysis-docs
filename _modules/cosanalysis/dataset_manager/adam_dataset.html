def fetch_adam_data(datasets=["ADSL", "ADPRO", "ADCL", "ADDI"], version=None, analysis_set="ISA"):
    """
    Fetch ADAM datasets from the DMP using the dmpy function fetch_adam_data.
    ADSL: Demographics data and TUP dates
    ADPRO: Weekly PROs
    ADCL: Clinical scores
    ADDI: Diary data

    :param datasets: Type of datasets we want to load. Must be among "ADSL", "ADPRO", "ADCL" and "ADDI".
    :type datasets: str or list of str

    :param version: Version of the data to load. Default is None.
    :type version: str, optional

    :param analysis_set: Either ISA or ESA (identification or evaluation dataset). Default is ISA.
    :type analysis_set: str, optional
    """
    analysis_set = analysis_set.upper()
    if analysis_set == "IAS":
        analysis_set = "ISA"

    if isinstance(datasets, str):
        datasets = [datasets]

    valid_datasets = ["ADSL", "ADPRO", "ADCL", "ADDI"]
    invalid_datasets = np.setdiff1d(datasets, valid_datasets)
    if len(invalid_datasets) > 0:
        raise ValueError(f"""Invalid argument 'dataset'. 
                        It must be a list included in {valid_datasets}.""")

    adam_datasets = {}

    # Fetch datasets
    print("Fetch ADSL dataset......")
    adsl_data = dmpy.fetch_adam_data(
        study_id, 'ADSL', version)    # demographics
    adsl_data = adsl_data[adsl_data.ANALYSISSET == analysis_set]

    # Fix invalid tup_dates
    if version is None:
        version_date = date.today()
    else:
        versions = dmpy.list_study_versions(study_id)
        version_date = pd.to_datetime(
            versions[versions.tag == version].created.iloc[0]).date()

    adsl_data.loc[pd.to_datetime(
        adsl_data.TUP1DT).dt.date >= version_date, "TUP1DT"] = pd.NaT
    adsl_data.loc[pd.to_datetime(
        adsl_data.TUP2DT).dt.date >= version_date, "TUP2DT"] = pd.NaT
    adsl_data.loc[pd.to_datetime(
        adsl_data.TUP3DT).dt.date >= version_date, "TUP3DT"] = pd.NaT
    adsl_data.loc[pd.to_datetime(
        adsl_data.TUP4DT).dt.date >= version_date, "TUP4DT"] = pd.NaT

    if "ADSL" in datasets:
        adsl_data.sort_values(by="USUBJID", inplace=True)
        adsl_data.reset_index(drop=True, inplace=True)
        adam_datasets["ADSL"] = adsl_data.copy()

    for d in [d for d in datasets if d != "ADSL"]:
        print(f"Fetch {d} dataset......")
        # Fetch dataset from the DMP
        dataset = dmpy.fetch_adam_data(study_id, d, version)
        # Keep only IAS data
        dataset = dataset[dataset.ANALYSISSET == analysis_set].copy()
        # Add COHORT column
        dataset = adsl_data[["USUBJID", "COHORT"]].merge(
            dataset, how="right")
        # To numeric
        dataset["AVAL"] = pd.to_numeric(dataset.AVAL, errors="coerce")
        dataset.sort_values(by=["USUBJID", "AVISIT", "ADT"], inplace=True)
        dataset.reset_index(drop=True, inplace=True)

        adam_datasets[d] = dataset.copy()

    return adam_datasets
