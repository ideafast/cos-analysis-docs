def read_adam_dataset(
    data_type,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    cohorts=None,
):
    """
    Reads adam dataset that is saved on the DMP for a given version, filtering by cohorts, participants and analysis set.

    :param data_type: The type of the data we want to load.
        Must be one of : ["ADDI", "ADSL", "ADCL", "ADPRO"]
    :type data_type: str

    :param version: Specifies the version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: The analysis set to use ('ias' or 'eas'). Default is 'ias'.
    :type analysis_set: str, optional

    :param participants: A list of participant IDs to include in the dataset. Default is None, all participants for the dataset (ias or eas) are loaded.
    :type participants: list of str, optional

    :param ias_dataset: Specifies the dataset of the identification set to load. Must be one of 'Train', 'Test', or 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: A list of cohorts to load. Default is None, all cohorts are loaded.
    :type cohorts: list of str, optional

    :return: A DataFrame containing the filtered dataset.
    :rtype: pandas.DataFrame

    :raises ValueError: If no file corresponding to the given parameters is found.

    """
    cohorts = check_cohorts(cohorts)
    data_type = check_valid_adam(data_type)
    analysis_set = check_valid_dataset(analysis_set)
    ias_dataset = check_valid_ias_dataset(ias_dataset)
    participants = check_participants(participants)

    adam_dataset = read_dataset(
        data_type, version, imputation, analysis_set)

    # Filter cohorts
    if cohorts is not None:
        adam_dataset = adam_dataset[
            adam_dataset.COHORT.isin(cohorts)
        ]

    # Filter participants
    if participants is not None:
        adam_dataset = adam_dataset[
            adam_dataset.USUBJID.isin(participants)
        ]

    # Filter ias_dataset
    elif (analysis_set == "ias") and (ias_dataset in ["Train", "Test"]):
        if data_type == "ADDI":
            ias_dataset = ias_dataset.upper()
        adam_dataset = adam_dataset[
            adam_dataset.IASSET == ias_dataset
        ]

    adam_dataset.rename(columns={"USUBJID": "subject_id"}, inplace=True)
    return adam_dataset.reset_index(drop=True).copy()

def load_diaries(
    concept,
    window_size,
    change_from_baseline=False,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    cohorts=None,
    agg_daily="mean"
):
    """
    Load diaries related to a specific study concept and a defined window size, with the option to account for changes from baselines, and select the participants or cohorts to include.

    :param concept: Concept to load. Must be one of 'F' (fatigue), 'SQ' (sleep quality), 'DS' (daytime sleepiness), 'all' (to load all three concepts), 'confounders' to load confounders (Anxiousness, Mood, Pain).
    :type concept: str

    :param window_size: The timing of the questions that we want to load. Must be one of 'intra_daily', 'daily'.
    :type window_size: str

    :param change_from_baseline: If True, we compute the changes from baselines, otherwise we load the answers of the patients. Default is False.
    :type change_from_baseline: bool, optional

    :param version: Specifies the version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :param agg_daily: function to aggregate diaries daily with. Default is "mean".
    :type agg_daily: str, optional

    :return: A DataFrame with diaries corresponding to the chosen parameters.
    :rtype: pandas.DataFrame

    :raises ValueError: If the parameters are not correct.
    """

    mapping_concept_questions = {
        "F": [
            "overall_fatigue_vas",
            "physical_fatigue_vas",
            "mental_fatigue_vas",
        ],
        "SQ": ["sleep_quality_vas"],
        "DS": ["KSS"],
        "confounders": ["anxiousness_vas", "mood_vas", "pain_vas"],
        "all": [
            "overall_fatigue_vas",
            "physical_fatigue_vas",
            "mental_fatigue_vas",
            "KSS",
            "sleep_quality_vas"
        ]
    }

    if concept not in mapping_concept_questions.keys():
        raise ValueError(
            f"Invalid concept: {concept}. Must be one of 'F', 'SQ', 'DS', 'all' and 'confounders'."
        )

    questions = mapping_concept_questions[concept]

    diaries = format_diaries(
        questions,
        window_size,
        change_from_baseline,
        version,
        imputation,
        analysis_set,
        participants,
        ias_dataset,
        cohorts,
        agg_daily
    )

    return diaries.copy()

def format_diaries(
    questions,
    window_size,
    change_from_baseline=False,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    cohorts=None,
    agg_daily="mean"
):
    """
    Load diaries choosing specific questions and with a defined window size, and the option to account for changes from baselines, and select the participants or cohorts to include.

    :param questions: List of questions to load. Each must be one of "overall_fatigue_vas", "physical_fatigue_vas", "mental_fatigue_vas", "sleep_quality_vas", "KSS", "PGIS-F", "PGIC-F", "PGIS-SQ", "PGIC-SQ", "PGIS-DS", "PGIC-DS", "anxiousness_vas", "mood_vas", "pain_vas".
    :type questions: list of str

    :param window_size: The timing of the questions that we want to load. Must be one of 'intra_daily', 'daily'.
    :type window_size: str

    :param change_from_baseline: If True, we compute the changes from baselines, otherwise we load the answers of the patients. Default is False.
    :type change_from_baseline: bool, optional

    :param version: Specifies the version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :param agg_daily: function to aggregate diaries daily with. Default is "mean".
    :type agg_daily: str, optional

    :return: A DataFrame with diaries corresponding to the chosen parameters.
    :rtype: pandas.DataFrame

    :raises ValueError: If the parameters are not correct.
    """

    if not isinstance(change_from_baseline, bool):
        raise ValueError("Change from baseline must be a boolean.")

    questions_dict = get_questions_dict()

    # Keep only valid questions
    valid_questions = list(questions_dict.keys())

    invalid_questions = [q for q in questions if q not in valid_questions]
    questions = [q for q in questions if q in valid_questions]

    if len(invalid_questions) > 0:
        raise ValueError(
            f"Questions {invalid_questions} are invalid. ",
            f"Valid questions are : {valid_questions}"
        )

    if len(questions) == 0:
        raise ValueError(
            f"No valid questions. Valid questions are : {valid_questions}")

    if window_size not in ["intra_daily", "daily"]:
        raise ValueError(
            f"Invalid window size: {window_size}. "
            "Must be one of 'intra_daily', 'daily'."
        )

    diary_questions = [questions_dict[question]["ADAM_dataset_name"]
                       for question in questions]

    diary = read_adam_dataset(
        data_type="ADDI",
        version=version,
        imputation=imputation,
        analysis_set=analysis_set,
        participants=participants,
        ias_dataset=ias_dataset,
        cohorts=cohorts,
    )

    # Filter questions
    diary = diary[diary.PARAMCD.isin(diary_questions)]

    # Pivot dataframe
    pivoted_diary_df = diary.pivot_table(
        index=["subject_id", "ADT", "TUPNUM", "TIMING", "ATM"],
        columns="PARAMCD",
        values="AVAL",
        aggfunc="first",
    ).reset_index()
    pivoted_diary_df.columns.name = None

    # Keep only the questions chosen
    columns_di = ["subject_id", "ADT", "ATM",
                  "TUPNUM", "TIMING"] + diary_questions
    diaries = pivoted_diary_df[columns_di].copy()

    # Remove NaNs rows
    diaries = diaries.dropna(subset=diary_questions, how="all")

    # Rename diary questions for better clarity
    rename_dict = dict(zip(diary_questions, questions))
    diaries.rename(columns=rename_dict, inplace=True)

    # Combine time columns
    diaries["ADT"] = pd.to_datetime(diaries["ADT"]).dt.date
    diaries["ATM"] = pd.to_datetime(
        diaries["ATM"], format="%H:%M:%S.%f").dt.time
    diaries["Time"] = pd.to_datetime(
        diaries.apply(
            lambda row: datetime.datetime.combine(row["ADT"], row["ATM"]), axis=1
        )
    )

    # Replace TUPNUM by tup
    diaries["tup"] = diaries["TUPNUM"].apply(map_tup)

    # Get tup dates
    tup_dates = get_tup_dates(version=version, analysis_set=analysis_set)

    # Change tup if needed
    diaries = add_tups(diaries, tup_dates)

    # Remove rows with tup=-1
    diaries = diaries.loc[diaries["tup"] != -1]

    diaries = diaries[
        (
            (diaries["TIMING"] == "M")
            & (diaries["Time"].dt.hour >= 6)
            & (diaries["Time"].dt.hour <= 11)
        )
        | (
            (diaries["TIMING"] == "A")
            & (diaries["Time"].dt.hour >= 12)
            & (diaries["Time"].dt.hour <= 16)
        )
        | (
            (diaries["TIMING"] == "E")
            & (diaries["Time"].dt.hour >= 17)
            & (diaries["Time"].dt.hour <= 23)
        )
        | (diaries["TIMING"] == "daily")
    ]

    diaries = diaries.sort_values(by=["subject_id", "Time"])
    diaries = diaries.drop_duplicates(
        subset=["subject_id", "ADT", "TIMING"], keep="first"
    )

    diaries = diaries.drop(columns=["ADT", "ATM", "TUPNUM"])

    # Convert questions to numeric
    for col in questions:
        diaries[col] = pd.to_numeric(diaries[col])

    if window_size == "daily":
        diaries = get_daily_diaries(diaries, agg_daily)
        diaries["TIMING"] = "daily"
        # Make sure all daily are in the same row
        diaries = diaries.groupby(["subject_id", "Time", 
                                   "tup", "TIMING"]
                                 ).first().reset_index()

    else:
        diaries["TIMING"] = diaries["TIMING"].replace("daily", "E")
        # Make sure all "E" are in the same row
        diaries = diaries.groupby(["subject_id",
                                   "tup", "TIMING"]
                                 ).first().reset_index()
    
    if change_from_baseline:
        baselines = get_baselines(diaries)
        diaries = diaries.merge(
            baselines.reset_index(), on=["subject_id", "TIMING"], how="right"
        )
        diaries = diaries.loc[diaries["tup"] != 1]

        for question in questions:
            diaries[f"{question}_change"] = (
                diaries[question] - diaries[f"{question}_baseline"]
            )
        diaries = diaries.drop(columns=[question for question in questions])

        questions = [f"{question}_baseline" for question in questions] + [
            f"{question}_change" for question in questions
        ]
        diaries = diaries.sort_values(by=["subject_id", "Time"])

    # Add tup day
    diaries = add_tup_day(diaries, tup_dates)

    # Change column order
    col = ["subject_id", "Time", "tup", "tup_day", "TIMING"] + questions
    diaries = diaries[col]

    diaries = diaries.reset_index(drop=True).copy()

    return diaries

def get_daily_diaries(diary, agg_daily):
    """
    Get the average of morning, afternoon, or evening diaries answered for each subject and day
    only if exactly 3 diaries were answered.

    :param diary: initial diary with 3 times a day answers.
    :type diary: pandas.DataFrame

    :param agg_daily: function to aggregate diaries daily with
    :type agg_daily: str

    :return: A DataFrame with the daily aggregated diaries.
    :rtype: pandas.DataFrame
    """
    diary = diary.copy()

    diary["Time"] = diary["Time"].dt.date

    questions_dict = get_questions_dict()

    agg_questions = [q for q in questions_dict if q in diary.columns
                     and questions_dict[q]["questions_per_day"] == 3]

    daily_questions = [q for q in questions_dict if q in diary.columns
                       and q not in agg_questions]

    daily_diary = diary.drop(columns=agg_questions).groupby(["subject_id", "Time", "tup"],
                                                            as_index=False).max()

    if agg_questions:
        # Count the number of answers for each subject and date and filter with only 3 answers
        valid_counts = (diary[diary["TIMING"].isin(['M', 'A', 'E'])]
                        .groupby(["subject_id", "Time"])
                        .filter(lambda x: len(x) == 3))

        diary = (valid_counts.groupby(["subject_id", "Time", "tup"])
                 .agg({col: agg_daily for col in agg_questions})
                 .round(1)
                 .reset_index())

        diary = diary.merge(daily_diary, how='outer')

    diary.dropna(how="all", subset=agg_questions +
                 daily_questions, inplace=True)

    return diary.copy()

def get_baselines(filtered_diary):
    """
    Get the baselines from a diary that is already filtered with its window size.

    :param filtered_diaries: DataFrame with either daily or intra_daily diaries.
    :type filtered_diaries: pandas.DataFrame

    :return: DataFrame with the following columns: subject_id, TIMING and the baselines (mean values of diary questions during TUP 1).
    :rtype: pandas.DataFrame
    """
    df_tup1 = filtered_diary[filtered_diary.tup == 1].copy()

    questions = [
        col
        for col in filtered_diary.columns
        if col not in ["subject_id", "Time", "tup", "tup_day", "ADT", "TIMING"]
    ]

    for col in questions:
        df_tup1[col] = pd.to_numeric(df_tup1[col])

    questions = ["subject_id", "TIMING"] + questions
    mean_results = (
        (df_tup1[questions].groupby(["subject_id", "TIMING"]).mean())
        .dropna(how="all")
        .round(1)
    )

    mean_results.columns = [
        f"{col}_baseline" for col in questions if col not in ["subject_id", "TIMING"]
    ]

    return mean_results.copy()

def get_demographics(version=None, imputation=False, analysis_set="ias", participants=None, ias_dataset="Train", cohorts=None):
    """
    Load demographics (COHORT, AGE, GENDER, BMI and TIMESINCEDIAG) for each subject.

    :param version: version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :return: A DataFrame containing the following columns : ["subject_id", "COHORT", "AGE", "GENDER", "BMI", "TIMESINCEDIAG"].
    :rtype: pandas.DataFrame
    """

    participants = read_adam_dataset(data_type="ADSL",
                                     version=version,
                                     imputation=imputation,
                                     analysis_set=analysis_set,
                                     participants=participants,
                                     ias_dataset=ias_dataset,
                                     cohorts=cohorts)

    columns = ["subject_id", "COHORT", "AGE", "GENDER", "BMI", "TIMESINCEDIAG"]

    demographics = (
        participants[columns].sort_values(
            by="subject_id").reset_index(drop=True)
    )
    for col in ["AGE", "BMI", "TIMESINCEDIAG"]:
        demographics[col] = pd.to_numeric(demographics[col])

    return demographics.copy()

def get_longitudinal_covariates(dataset):
    """
    Load longitudinal covariates (Day_Type, Season) for each subject_id, Date.

    :param dataset: DataFrame containing "subject_id" and "Date" columns.
    :type dataset: pandas.DataFrame

    :return: A DataFrame with the following columns ["subject_id", "Date", "Season", "Day_Type"]
    :rtype: pandas.DataFrame
    """
    dataset = dataset.copy()
    longitudinal_covariates = (
        dataset[["subject_id", "Date"]].drop_duplicates().reset_index(drop=True)
    )

    # Weekday
    longitudinal_covariates["Day_Type"] = pd.to_datetime(
        longitudinal_covariates["Date"]
    ).dt.dayofweek.apply(lambda x: "Weekday" if x < 5 else "Weekend")

    # Season
    longitudinal_covariates["Season"] = longitudinal_covariates["Date"].apply(
        assign_season
    )

    return longitudinal_covariates.copy()

def assign_season(date):
    """
    Find the season corresponding to a date.

    :param date: Date to assign a season to.
    :type date: datetime

    :return: Season (Spring, Summer, Fall or Winter)
    :rtype: str
    """
    month = date.month
    day = date.day

    if (month == 3 and day >= 21) or (month in [4, 5]) or (month == 6 and day <= 20):
        return "Spring"
    elif (month == 6 and day >= 21) or (month in [7, 8]) or (month == 9 and day <= 20):
        return "Summer"
    elif (
        (month == 9 and day >= 21) or (month in [
            10, 11]) or (month == 12 and day <= 20)
    ):
        return "Fall"
    else:
        return "Winter"

def map_tup(tup_value):
    """
    Associates a string containing a TUP statement with the corresponding integer. ex: TUP 1 -> 1

    :param tup_value: String to be analyzed.
    :type tup_value: str

    :return: An integer corresponding to the number found in the string (1 to 4) or None if no match is found.
    :rtype: int, optional
    """
    for i in range(1, 5):
        if f"TUP {i}" in tup_value:
            return i
    return None  # If there is no match

def get_baseline_PROs(version=None, imputation=False, analysis_set="ias", participants=None, ias_dataset="Train", cohorts=None):
    """
    Get the baseline (value at Start of TUP 1) of PROs (FACITF-SCORE, KSS and SLPD4). This functions calls get_tup_PROs function.

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :return: DataFrame with the following columns: ["subject_id, "FACITF-SCORE_baseline_7days", "KSS_baseline_7days", "SLPD4_baseline_7days"]
    :rtype: pandas.DataFrame
    """

    questions = ["FACITF-SCORE", "KSS", "SLPD4"]
    pros = get_tup_PROs(questions, version, imputation,
                        analysis_set, participants, ias_dataset, cohorts)
    pros_tup1 = pros[pros.ATUP == 'Start of TUP 1'].copy()

    results = pros_tup1[['subject_id'] + questions].copy()

    for col in questions:
        results.rename(columns={col: f"{col}_baseline_7days"}, inplace=True)

    return results.copy()

def get_disease_score(version=None, imputation=False, analysis_set="ias", participants=None, ias_dataset="Train", cohorts=None):
    """
    Get the diseases score for each patients of specified cohorts.

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :return: DataFrame with disease scores corresponding to each subject_id and tup.
    :rtype: pandas.DataFrame

    :raises ValueError: If cohorts argument is invalid.
    """

    clinical_df = read_adam_dataset(data_type="ADCL",
                                    version=version,
                                    imputation=imputation,
                                    analysis_set=analysis_set,
                                    participants=participants,
                                    ias_dataset=ias_dataset,
                                    cohorts=cohorts)

    clinical_df["tup"] = clinical_df["TUPNUM"].apply(map_tup)

    clinical_df = clinical_df.rename(columns={"USUBJID": "subject_id"})

    filtered_df = clinical_df.set_index(["subject_id", "tup"])[
        ["PARAMCD", "AVAL"]]

    filtered_df = (
        filtered_df.rename(
            columns={"PARAMCD": "Disease_test", "AVAL": "score"})
        .sort_values(["subject_id", "tup"])
        .reset_index()
    )

    filtered_df = filtered_df[filtered_df.Disease_test.str.endswith('SCORE')]

    return filtered_df.copy()

def get_tup_PROs(questions=None, version=None, imputation=False, analysis_set="ias", participants=None, ias_dataset="Train", cohorts=None):
    """
    This function retrieves PROs data from the ADAM dataset ADPRO.
    It allows filtering by version.

    :param questions: List of pros to load. Default is None. Each must be one of "FACITF-SCORE", "SLPD4", "KSS", "fVAS", "sqVAS", "PGIS-F", "PGIC-F", "PGIS-SQ", "PGIC-SQ", "PGIS-DS", "PGIC-DS".
    :type questions: list of str

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :param participants: List of participant to consider. Default is None, we load all participant from the given dataset (ias or eas).
    :type participants: str or list of str, optional

    :param ias_dataset: If participants is None, we load the Train set, the Test set or both. Must be one of 'Train', 'Test' and 'all'. Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: The list of the cohorts to consider. Default is None, we consider them all.
    :type cohorts: list of str, optional

    :return: DataFrame with TUP-based PROs.
    :rtype: pandas.DataFrame

    """

    analysis_set = check_valid_dataset(analysis_set)
    ias_dataset = check_valid_ias_dataset(ias_dataset)

    valid_questions = [
        "FACITF-SCORE", "SLPD4", "KSS", "fVAS", "sqVAS",
        "PGIS-F", "PGIC-F", "PGIS-SQ", "PGIC-SQ", "PGIS-DS", "PGIC-DS"
    ]

    if questions is None:
        questions = ["FACITF-SCORE", "SLPD4", "KSS", "fVAS", "sqVAS"]

    invalid_questions = [q for q in questions if q not in valid_questions]

    if len(invalid_questions) > 0:
        raise ValueError(
            f"Questions {invalid_questions} are invalid. ",
            f"Valid questions are : {valid_questions}"
        )

    if len(questions) == 0:
        raise ValueError(
            f"No valid questions. Valid questions are : {valid_questions}")

    pros = read_adam_dataset(data_type="ADPRO",
                             version=version,
                             imputation=imputation,
                             analysis_set=analysis_set,
                             participants=participants,
                             ias_dataset=ias_dataset,
                             cohorts=cohorts)

    pros = pros[pros.PARAMCD.isin(questions)]

    pivoted_pros_df = pros.pivot_table(
        index=["subject_id", "TUPNUM", "ATUP"],
        columns="PARAMCD",
        values="AVAL",
        aggfunc="first",
    ).reset_index()
    pivoted_pros_df.columns.name = None

    pivoted_pros_df["tup"] = pivoted_pros_df["TUPNUM"].apply(map_tup)

    # Keep only the useful columns
    columns_pros = ["subject_id", "tup", "ATUP"] + questions
    pros_df = pivoted_pros_df[columns_pros].copy()
    pros_df = pros_df.rename(columns={"ADT": "Date"})

    return pros_df.copy()

def expand_row(row):
    """
    Get a DataFrame with a row for each day between the start_date and end_date. Useful only if TIMING is daily.

    :param row: Row containing "subject_id", "tup", "start_date" and "stop_date" information.
    :type row: pandas.Series

    :return: DataFrame with all dates in the range, containing the following columns ["subject_id", "Time", "TIMING", "tup", "tup_day"]
    :rtype: pandas.DataFrame
    """
    date_range = pd.date_range(row["start_date"], row["stop_date"])
    return pd.DataFrame(
        {
            "subject_id": row["subject_id"],
            "Time": date_range,
            "TIMING": "daily",
            "tup": row["tup"],
            "tup_day": range(0, len(date_range)),
        }
    )

def get_timings(subjects=None,
                timing="MAE",
                version=None,
                diaries=None,
                imputation=False,
                analysis_set="ias",
                ):
    """
    Get timings corresponding either to diaries, or to subjects for a specified timing (M, A, E, MAE, intra_daily or daily).

    :param subjects: List of subjects to consider. Default is None, we return None if diaries is also None, there is no incidence if diaries is not None.
    :type subjects: list of str, optional

    :param timing: timing to load if diaries is not None. Must be one of "M", "A", "E", "MAE", "intra_daily", "daily". Default is "MAE".
    :type timing: str, optional

    :param version: Version of the data to use. Default is None.
    :type version: str, optional

    :param diaries: diaries to extract timings from (extract the following columns : ["subject_id", "tup", "tup_day", "TIMING", "Time"])
    :type diaries: pandas.DataFrame

    :param imputation: Specifies if the dataset should include imputation data. Default is False.
    :type imputation: bool, optional

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :return: DataFrame with timings (contains the following columns : ["subject_id", "tup", "tup_day", "TIMING", "Time"])
    :rtype: pandas.DataFrame

    :raises ValueError: if timing not in ["M", "A", "E", "MAE", "intra_daily", "daily"], or if subjects are not valid.
    """
    analysis_set = check_valid_dataset(analysis_set)

    if diaries is not None:
        timings = diaries[["subject_id", "tup",
                           "tup_day", "TIMING", "Time"]].copy()
        timings.Time = pd.to_datetime(timings.Time)
        return timings

    subjects = check_participants(subjects)

    if subjects is None or len(subjects) == 0:
        return None

    if timing not in ["M", "A", "E", "MAE", "intra_daily", "daily"]:
        raise ValueError(
            "timing must be {'M', 'A', 'E', 'MAE', 'intra_daily', 'daily'}"
        )

    tup_dates = get_tup_dates(version=version, analysis_set=analysis_set)
    tup_dates = tup_dates[tup_dates.subject_id.isin(subjects)].copy()

    if timing == "daily":
        timings = pd.concat(
            [expand_row(row) for _, row in tup_dates.iterrows()], ignore_index=True
        )

    else:
        timings = read_adam_dataset(data_type="ADDI", version=version,
                                    imputation=imputation, analysis_set=analysis_set,
                                    participants=subjects)

        timings = timings.rename(columns={"USUBJID": "subject_id"})

        if timings.empty:
            return timings

        # Combine time columns
        timings["ADT"] = pd.to_datetime(timings["ADT"]).dt.date
        timings["ATM"] = pd.to_datetime(
            timings["ATM"], format="%H:%M:%S.%f").dt.time
        timings["Time"] = pd.to_datetime(
            timings.apply(
                lambda row: datetime.datetime.combine(row["ADT"], row["ATM"]), axis=1
            )
        )

        timings = timings.drop_duplicates(
            subset=["subject_id", "TUPNUM", "TIMING", "Time"], keep="first"
        )[["subject_id", "Time", "TUPNUM", "TIMING"]]
        timings = timings[timings.TIMING.isin(["M", "A", "E"])]

        # Map TUPs
        timings["tup"] = timings["TUPNUM"].apply(map_tup)

        # Change tup if needed
        timings = add_tups(timings, tup_dates)

        if timing in ["M", "A", "E"]:
            timings = timings[timings.TIMING == timing]

    # Add tup day
    timings = add_tup_day(timings, tup_dates)

    timings.sort_values(["subject_id", "Time"], inplace=True)
    timings = timings[["subject_id", "tup", "tup_day", "TIMING", "Time"]].reset_index(
        drop=True
    )

    return timings
