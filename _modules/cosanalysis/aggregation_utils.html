def aggregate_periodic_data(
    data,
    time_header,
    window_start_header,
    window_stop_header,
    categorical_columns,
    continuous_columns,
    period="D",
    freq=None,
    thresh=0.75,
    offset=None,
    origin=None,
    by=None,
    statistics=None,
    additional_functions=None,
):
    """
    Aggregate dataset by period.

    This function aggregates data into specified periods, such as daily, weekly, etc.,
    and computes statistical functions on both categorical and continuous columns.

    offset, period, and origin defined here:
    https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases

    :param data: Input DataFrame to be aggregated.
    :type data: pandas.DataFrame

    :param time_header: Name of the column containing the time values.
    :type time_header: str

    :param window_start_header: Name of the column where the start time of the window will be stored.
    :type window_start_header: str

    :param window_stop_header: Name of the column where the stop time of the window will be stored.
    :type window_stop_header: str

    :param categorical_columns: List of column names representing categorical data for aggregation.
    :type categorical_columns: list of str

    :param continuous_columns: List of column names representing continuous data for aggregation.
    :type categorical_columns: list of str

    :param period: The period for aggregation (e.g., "D" for day, "W" for week, etc.). Default is "D".
    :type period: str

    :param freq: Frequency of data, used to filter groups based on coverage. Default is None.
    :type freq: float, optional

    :param thresh: Proportion of non-missing values required for aggregation to occur in each group. Default is 0.75.
    :type thresh: float, optional

    :param offset: Offset to be applied to the period (e.g., "+1D").
    :type offset: str, optional

    :param origin: The reference date for the aggregation (e.g., 'start', 'epoch').
    :type origin: str, optional

    :param by: Column or columns by which to group the data.
    :type by: str or list of str, optional

    :param statistics: List of statistical functions to apply (e.g., "mean", "std").
    :type statistics: list of str, optional

    :param additional_functions: Dictionary of custom functions to apply to specific columns.
    :type additional_functions: dict, optional

    :return: Aggregated DataFrame with statistical summaries for each group.
    :rtype: pandas.DataFrame
    """

    if additional_functions is None:
        additional_functions = {}

    if by is None:
        by = []
    elif isinstance(by, str):
        by = [by]

    data = data.copy()
    data[time_header] = pd.to_datetime(data[time_header])

    grouper = by + [
        pd.Grouper(key=time_header, freq=period, offset=offset, origin=origin)
    ]

    # filter groups based on coverage
    if freq is not None:
        period_seconds = pd.to_timedelta(to_offset(period)).total_seconds()
        max_nb_values = period_seconds * freq

        threshold_value = thresh * max_nb_values
        data = data.groupby(grouper).filter(
            lambda x: len(x) >= threshold_value)

    # aggregate data
    group = data.groupby(grouper)
    data_agg = aggregate_data(
        group,
        categorical_columns,
        continuous_columns,
        stats=statistics,
        custom_functions=additional_functions,
    )

    # Add start and stop window columns
    if (window_start_header not in data_agg.columns) and (window_stop_header not in data_agg.columns):
        data_agg.rename(
            columns={time_header: window_start_header}, inplace=True)
        data_agg.insert(
            len(by) + 1,
            window_stop_header,
            data_agg[window_start_header] + pd.to_timedelta(period),
        )

    return data_agg

def aggregate_windowed_data(
    data,
    time_header,
    df_windows,
    window_start_header,
    window_stop_header,
    categorical_columns,
    continuous_columns,
    freq=None,
    thresh=0.75,
    by=None,
    statistics=None,
    additional_functions=None,
):
    """
    Aggregate dataset by windows defined in df_windows DataFrame.

    This function aggregates the data based on predefined start and stop windows from the df_windows DataFrame.

    :param data: Input DataFrame containing the data to be aggregated.
    :type data: pandas.DataFrame

    :param time_header: Name of the column containing the time values.
    :type time_header: str

    :param df_windows: DataFrame containing the predefined windows with start and stop times for aggregation.
    :type df_windows: pandas.DataFrame

    :param window_start_header: Name of the column containing the start time of the window.
    :type window_start_header: str

    :param window_stop_header: Name of the column containing the stop time of the window.
    :type window_stop_header: str


    :param categorical_columns: List of categorical columns for which statistical functions are applied.
    :type categorical_columns: list of str

    :param continuous_columns: List of continuous columns for which statistical functions are applied.
    :type continuous_columns: list of str

    :param freq: Frequency of data, used to filter out groups that don't meet the coverage threshold.
    :type freq: float, optional

    :param thresh: Proportion of non-missing values required for aggregation to occur in each group. Default is 0.75.
    :type thresh: float, optional

    :param by: Columns by which to group the data (e.g., "subject_id").
    :type by: str or list of str, optional

    :param statistics: List of statistical functions to apply (e.g., "mean", "std").
    :type statistics: list of str, optional

    :param additional_functions: Dictionary of additional custom aggregation functions to apply.
    :type additional_functions: dict, optional

    :return: Aggregated DataFrame with statistical summaries for each window.
    :rtype: pandas.DataFrame
    """

    if additional_functions is None:
        additional_functions = {}

    if by is None:
        by = []
    elif isinstance(by, str):
        by = [by]

    # get windowed data
    data_windowed = get_aggregation_windows(
        data, time_header, df_windows, window_start_header, window_stop_header, by=by
    )

    grouper = by + [window_start_header, window_stop_header]

    # filter groups based on coverage
    if freq is not None:
        window_seconds = (
            data_windowed.iloc[0][window_stop_header]
            - data_windowed.iloc[0][window_start_header]
        ).total_seconds()
        max_nb_values = window_seconds * freq

        threshold_value = thresh * max_nb_values
        data_windowed = data_windowed.groupby(grouper).filter(
            lambda x: len(x) >= threshold_value
        )

    # aggregate data
    group = data_windowed.groupby(grouper)
    data_agg = aggregate_data(
        group,
        categorical_columns,
        continuous_columns,
        stats=statistics,
        custom_functions=additional_functions,
    )

    return data_agg.copy()

def get_aggregation_windows(
    data,
    time_header,
    df_times,
    start_header,
    stop_header,
    by=None,
):
    """
    Map dataset times to start and stop times defined in df_times.

    This function assigns each row in the data to a specific window defined by the start and stop times from the df_times DataFrame.

    :param data: Input DataFrame to be aggregated.
    :type data: pandas.DataFrame

    :param time_header: Name of the column containing the time values in the data.
    :type time_header: str

    :param df_times: DataFrame containing the windows with start and stop times.
    :type df_times: pandas.DataFrame

    :param start_header: Name of the column containing the start time of the window.
    :type start_header: str

    :param stop_header: Name of the column containing the stop time of the window.
    :type stop_header: str

    :param by: Columns by which to group the data (e.g., "subject_id").
    :type by: list of str, optional

    :return: DataFrame where each row is assigned to a window based on the time it falls into.
    :rtype: pandas.DataFrame

    :raises ValueError: If the input DataFrames don't contain the required columns, or if the datetime columns don't share the same timezones.
    :raises RuntimeError: If get_aggregation_windows is not properly executed.
    """

    if time_header not in data.columns:
        raise ValueError(f"data must contain a column {time_header}.")
    if start_header not in df_times.columns:
        raise ValueError(f"df_times must contain a column {start_header}.")
    if stop_header not in df_times.columns:
        raise ValueError(f"df_times must contain a column {stop_header}.")

    if by is not None:
        by = [col for col in by if col in df_times.columns and col in data.columns]
    else:
        by = []

    data = data.copy()

    data[time_header] = pd.to_datetime(data[time_header])
    df_times[start_header] = pd.to_datetime(df_times[start_header])
    df_times[stop_header] = pd.to_datetime(df_times[stop_header])

    df_times.dropna(subset=[start_header, stop_header], inplace=True)

    timezones = [
        data[time_header].dt.tz,
        df_times[start_header].dt.tz,
        df_times[stop_header].dt.tz,
    ]
    if len(set(timezones)) != 1:
        raise ValueError(
            f"columns {time_header}, {start_header}, {stop_header} must share the same timezone."
        )

    epsilon = timedelta(microseconds=1)

    try:
        merged_data = pd.merge(data, df_times, on=by, how="left")
        data_windowed = merged_data[
            (merged_data[time_header] >= merged_data[start_header])
            & (merged_data[time_header] <= merged_data[stop_header] - epsilon)
        ]
    except:
        raise RuntimeError("get_aggregation_windows was not properly executed")

    return data_windowed

def aggregate_data(
    group,
    categorical_columns,
    continuous_columns,
    stats=None,
    custom_functions={},
):
    """
    Aggregate data by applying statistical functions to categorical and continuous columns.

    This function computes statistical summaries for continuous and categorical columns in each group.

    :param group: The grouped DataFrame to aggregate.
    :type group: GroupBy object

    :param categorical_columns: List of categorical columns for which aggregation is to be performed.
    :type categorical_columns: list of str

    :param continuous_columns: List of continuous columns for which aggregation is to be performed.
    :type continuous_columns: list of str

    :param stats: List of statistical functions to apply (e.g., "mean", "max").
    :type stats: list of str, optional

    :param custom_functions: Dictionary of custom functions to apply to specific columns.
    :type custom_functions: dict, optional


    :return: Aggregated DataFrame containing the computed statistics.
    :rtype: pandas.DataFrame
    """

    allowed_continuous_stats = ["mean", "std",
                                "var", "max", "min", "25p", "50p", "75p"]
    allowed_categorical_stats = ["mode"]

    if stats is None:
        continuous_stats = allowed_continuous_stats
        categorical_stats = allowed_categorical_stats
    else:
        continuous_stats = np.intersect1d(allowed_continuous_stats, stats)
        categorical_stats = np.intersect1d(allowed_categorical_stats, stats)

    data_agg_list = []

    if len(continuous_columns) > 0 and len(continuous_stats) > 0:
        data_agg_ = (
            group[continuous_columns]
            .agg(get_stats_functions(continuous_stats))
            .reset_index()
        )
        data_agg_list.append(data_agg_)

    if len(categorical_columns) > 0 and len(categorical_stats) > 0:
        data_agg_ = (
            group[categorical_columns]
            .agg(get_stats_functions(categorical_stats))
            .reset_index()
        )
        data_agg_list.append(data_agg_)

    if len(custom_functions) > 0:
        data_agg_ = group.agg(custom_functions).reset_index()
        data_agg_list.append(data_agg_)

    if len(data_agg_list) == 0:
        return pd.DataFrame()

    data_agg = reduce(
        lambda left, right: pd.merge(left, right, how="outer"), data_agg_list
    )
    data_agg.columns = ["_".join(h) if h[1] != "" else h[0]
                        for h in data_agg.columns]

    return data_agg

def get_stats_functions(stats):
    """
    Map statistical function names to the corresponding function implementations.

    This function returns a list of tuples where each tuple contains a statistical function name
    and the corresponding function to apply to the data.

    :param stats: List of statistical functions to map (e.g., "mean", "std").
    :type stats: list of str

    :return: List of tuples, each containing a function name and the corresponding function.
    :rtype: list of tuples
    """
    from scipy.stats import kurtosis, skew, iqr

    statistics = {
        "mean": "mean",
        "std": "std",
        "var": "var",
        "sum": "sum",
        "energy": lambda x: np.sqrt(np.mean(np.power(x[~np.isnan(x)], 2))),
        "kurtosis": lambda x: kurtosis(x[~np.isnan(x)]),
        "skew": lambda x: skew(x[~np.isnan(x)]),
        "iqr": lambda x: iqr(x[~np.isnan(x)]),
        "max": "max",
        "min": "min",
        "10p": lambda x: (
            np.nan if np.all(np.isnan(x)) else np.percentile(
                x[~np.isnan(x)], 10)
        ),
        "25p": lambda x: (
            np.nan if np.all(np.isnan(x)) else np.percentile(
                x[~np.isnan(x)], 25)
        ),
        "50p": "median",
        "75p": lambda x: (
            np.nan if np.all(np.isnan(x)) else np.percentile(
                x[~np.isnan(x)], 75)
        ),
        "90p": lambda x: (
            np.nan if np.all(np.isnan(x)) else np.percentile(
                x[~np.isnan(x)], 90)
        ),
        "deviation": lambda x: np.max(x[~np.isnan(x)]) - np.min(x[~np.isnan(x)]),
        "mode": lambda x: (
            pd.Series.mode(x[~pd.isna(x)])[
                0] if not x[~pd.isna(x)].empty else None
        ),
    }

    statistics_ = [(s, statistics[s]) for s in stats]
    return statistics_

def calculate_start_end(df_windows, window_length, setting):
    """
    Compute start and end times based on window length and setting ('before', 'after', 'around').

    :param df_windows: DataFrame containing the windows with times.
    :type df_windows: pandas.DataFrame

    :param window_length: Length of the window in hours.
    :type window_length: int

    :param setting: Setting for how the window is defined: 'before', 'after', or 'around'.
    :type setting: str

    :return: Start and end times for the windows as Pandas Series.
    :rtype: tuple of pandas.Series

    :raises ValueError: If the setting is not one of "before", "around", "after".
    """

    if setting == "before":
        start_time = df_windows.Time - timedelta(hours=window_length)
        end_time = df_windows.Time
    elif setting == "after":
        start_time = df_windows.Time
        end_time = df_windows.Time + timedelta(hours=window_length)
    elif setting == "around":
        start_time = df_windows.Time - timedelta(hours=window_length / 2)
        end_time = df_windows.Time + timedelta(hours=window_length / 2)
    else:
        raise ValueError(
            f"{setting} is not a valid setting. Must be one of 'before', 'around', 'after'."
        )

    start_time = pd.to_datetime(start_time)
    end_time = pd.to_datetime(end_time)

    return start_time, end_time

def check_period(period):
    """
    Validate the given period string.

    This function checks if the provided period is a valid period specification.

    :param period: The period string to validate (e.g., '2D', '1W', '5h').
    :type period: str

    :return: The validated period string, potentially with a prepended '1' if only an alias is provided.
    :rtype: str

    :raises ValueError:
        If the period is invalid.
    """

    valid_periods = r"^\d*(?:D|W|h|min)$"

    is_valid = bool(re.match(valid_periods, period))

    if not is_valid:
        raise ValueError(
            "Invalid period. Accepted units are "
            + "'D' (calendar day frequency)"
            + ", 'W' (weekly frequency)"
            + ", 'h' (hourly frequency)"
            + ", 'min' (minutly frequency).\n"
            + "Examples of periods: '2D', '7D', '1W', '2h', '30min'"
        )

    if period.isalpha():
        period = f"1{period}"

    return period

def is_daily_period(period):
    """
    Check if the given period represents a daily frequency.

    :param period: The period string (e.g., '2D', '1W').
    :type period: str

    :return: True if the period represents a daily frequency, False otherwise.
    :rtype: bool
    """
    daily = pd.to_timedelta(period).total_seconds() % (24 * 3600) == 0
    return daily
