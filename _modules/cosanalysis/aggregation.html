def get_device_type(device_dataset):
    """
    Identifies the device type based on the column names in the provided dataset.

    :param device_dataset: The dataset containing device data.
    :type device_dataset: pandas.DataFrame

    :return: The device type, inferred from column names.
    :rtype: str or None
    """
    device_type = None

    pattern = r"^(slp|mcr|mcl|cog|VIT|gva|HRV|sts)"

    for col in device_dataset.columns:
        match = re.match(pattern, col)
        if match:
            device_type = match.group(1)
            break
    return device_type

def aggregate_device_data_on_timings(
    device_dataset,
    timings,
    window_length=2,
    setting="around",
    statistics=None,
    thresh=0.75,
):
    """
    Aggregates device data based on specific time windows and other criteria,
    adjusting for periodicities and categorical classifications.

    :param device_dataset: The dataset containing device data to be aggregated.
    :type device_dataset: pandas.DataFrame

    :param timings: A dataframe containing the timing information used for aggregation.
    :type timings: pandas.DataFrame

    :param window_length: The length of the window (in hours) for data aggregation. Default is 2.
    :type window_length: int, optional

    :param setting: Defines whether to aggregate data "around" or "before" or "after" a given timing. Default is "around".
    :type setting: str, optional

    :param statistics: A list of statistics to apply to the continuous columns (e.g., ['mean', 'std']). Default is None, which uses ["mean", "std", "var", "max", "min", "25p", "50p", "75p", "mode"].
    :type statistics: list of str, optional

    :param thresh: The threshold for including data based on duration. Default is 0.75.
    :type thresh: float, optional

    :return: A dataframe with aggregated device data based on the specified time windows.
    :rtype: pandas.DataFrame

    :note:
        - Aggregates based on device type (e.g., "mcl", "sts", "VIT").
        - Incorporates additional functions and custom transformations based on the device type.
    """

    additional_functions = {}
    by = ["subject_id", "tup", "tup_day"]

    # device dataset
    device_dataset = device_dataset.copy()
    device_type = get_device_type(device_dataset)
    device_measures = get_device_measures(device_dataset)
    freq = freqs[device_type] if device_type in freqs.keys() else None

    all_device_cat_columns = [
        "VIT_PostureCategory_1min",
        "mcr_class",
    ]

    continuous_columns = [
        col
        for col in device_measures
        if col not in all_device_cat_columns
        and col
        not in [
            "sts_type",
            "mcl_class",
            "mcl_status",
            "mcl_duration",
            "mcl_movement_time",
            "mcl_category",
        ]
    ]

    categorical_columns = [
        col for col in device_measures if col in all_device_cat_columns
    ]

    if (
        (statistics is not None)
        and (len(categorical_columns) > 0)
        and ("mode" not in statistics)
    ):
        statistics.append("mode")

    # timings
    df_windows = timings.merge(
        device_dataset[["subject_id", "tup", "tup_day"]].drop_duplicates(), how="inner"
    )
    df_windows.Time = pd.to_datetime(df_windows.Time)

    if ("start_window" not in df_windows) and ("stop_window" not in df_windows):
        start_time, end_time = calculate_start_end(df_windows, window_length, setting)
        df_windows["start_window"] = start_time
        df_windows["stop_window"] = end_time

    if device_type == "mcl":
        device_dataset.loc[device_dataset["mcl_class"] == "transition", "mcl_class"] = (
            "transition_" + device_dataset["mcl_category"].astype(str)
        )
        device_dataset.drop(columns="mcl_category", inplace=True)
        device_dataset.drop(columns="mcl_status", inplace=True)

        by += ["mcl_class"]

        additional_functions = {
            "mcl_duration": ["sum"],
            "mcl_movement_time": ["sum"],
            "mcl_class": ["count"],
        }

    elif device_type == "sts":
        by += ["sts_type"]
        additional_functions = {"sts_type": ["count"]}

    data_agg = aggregate_windowed_data(
        data=device_dataset,
        time_header="start_timestamp",
        df_windows=df_windows,
        window_start_header="start_window",
        window_stop_header="stop_window",
        categorical_columns=categorical_columns,
        continuous_columns=continuous_columns,
        freq=freq,
        thresh=thresh,
        by=by,
        statistics=statistics,
        additional_functions=additional_functions,
    )
    data_agg = df_windows.merge(data_agg, how="inner")

    index = [col for col in data_agg.columns 
             if not col.startswith(device_type)]
    
    window_seconds = (df_windows["stop_window"].iloc[0] - df_windows["start_window"].iloc[0]).total_seconds()
    time_header = ["start_window", "stop_window"]

    if device_type == "sts":
        data_agg = data_agg.pivot_table(
            index=index, columns="sts_type", aggfunc="first"
        ).reset_index()

        data_agg.columns = [
            (
                f"sts_{sts_type}_{val.replace('sts_', '').replace('type_count', 'Events_count')}"
                if sts_type
                else val
            )
            for val, sts_type in data_agg.columns
        ]

        data_agg.sts_SitStand_Events_count = data_agg.sts_SitStand_Events_count.fillna(
            0
        )
        data_agg.sts_StandSit_Events_count = data_agg.sts_StandSit_Events_count.fillna(
            0
        )

    elif device_type == "mcl":
        data_agg.rename(
            columns={
                "mcl_duration_sum": "mcl_duration",
                "mcl_movement_time_sum": "mcl_movement_time",
            },
            inplace=True,
        )
        data_agg["mcl_total_duration"] = data_agg.groupby(["subject_id"] + time_header)[
            "mcl_duration"
        ].transform("sum")
        threshold_value = thresh * window_seconds
        data_agg = data_agg[data_agg.mcl_total_duration >= threshold_value].copy()
        data_agg.drop(columns=["mcl_total_duration"], inplace=True)

        data_agg = data_agg.pivot_table(
            index=index, columns="mcl_class", aggfunc="first"
        ).reset_index()

        data_agg.columns = [
            (
                f"mcl_{mcl_class}_{val.replace('mcl_', '').replace('class_count', 'Events_count')}"
                if mcl_class
                else val
            )
            for val, mcl_class in data_agg.columns
        ]

        to_drop = [
            col
            for col in data_agg.columns
            if col.endswith("Events_count") and not col.endswith("shift_Events_count")
        ]
        data_agg.drop(
            columns=[
                col
                for col in data_agg.columns
                if col.endswith("shift_movement_time") or col in to_drop
            ],
            inplace=True,
        )
        data_agg = data_agg.fillna(0)

        # Add class mode
        duration_columns = [c for c in data_agg.columns if c.endswith("_duration")]
        position = (
            data_agg[duration_columns]
            .idxmax(axis=1)
            .str.replace("_duration", "")
            .str.replace("mcl_", "")
        )
        data_agg.insert(len(index), "mcl_class", position)

        # reorganise
        transition_cols = [
            c for c in data_agg.columns if c.startswith("mcl_transition")
        ]
        data_agg = data_agg[
            [c for c in data_agg.columns if c not in transition_cols] + transition_cols
        ]

    return data_agg.copy()

def aggregate_device_data_on_frequency(
    device_dataset,
    period="D",
    statistics=None,
    thresh=0.75,
):
    """
    Aggregates device data based on a specific frequency period (e.g., daily, hourly, etc.).

    :param device_dataset: The dataset containing device data to be aggregated.
    :type device_dataset: pandas.DataFrame

    :param period: The frequency period for aggregation (e.g., 'D' for daily, 'H' for hourly).
    :type period: str, optional

    :param statistics: A list of statistics to apply to the continuous columns (e.g., ['mean', 'std']). Default is None, which uses ["mean", "std", "var", "max", "min", "25p", "50p", "75p", "mode"].
    :type statistics: list of str, optional

    :param thresh: The threshold for including data based on duration. Default is 0.75.
    :type thresh: float, optional

    :return: A dataframe with aggregated device data based on the specified frequency.
    :rtype: pandas.DataFrame

    :note:
        - Aggregates data based on device type (e.g., "mcl", "sts", "VIT").
        - Includes functionality to handle transitions, class aggregations, and event counts.
    """

    additional_functions = {}
    by = ["subject_id", "tup"]

    period = check_period(period)
    daily = is_daily_period(period)
    day_factor = daily and period != "1D"

    if not day_factor:
        by += ["tup_day"]

    # device dataset
    device_dataset = device_dataset.copy()
    device_type = get_device_type(device_dataset)
    device_measures = get_device_measures(device_dataset)
    freq = freqs[device_type] if device_type in freqs.keys() else None

    all_device_cat_columns = [
        "VIT_PostureCategory_1min",
        "mcr_class",
    ]

    continuous_columns = [
        col
        for col in device_measures
        if col not in all_device_cat_columns
        and col
        not in [
            "sts_type",
            "mcl_class",
            "mcl_status",
            "mcl_duration",
            "mcl_movement_time",
            "mcl_category",
        ]
    ]

    categorical_columns = [
        col for col in device_measures if col in all_device_cat_columns
    ]
    if (
        (statistics is not None)
        and (len(categorical_columns) > 0)
        and ("mode" not in statistics)
    ):
        statistics.append("mode")

    if device_type == "mcl":

        device_dataset.loc[device_dataset["mcl_class"] == "transition", "mcl_class"] = (
            "transition_" + device_dataset["mcl_category"].astype(str)
        )
        device_dataset.drop(columns="mcl_category", inplace=True)

        if daily:
            by += ["mcl_status"]
        else:
            device_dataset.drop(columns="mcl_status", inplace=True)
        by += ["mcl_class"]

        additional_functions = {
            "mcl_duration": ["sum"],
            "mcl_movement_time": ["sum"],
            "mcl_class": ["count"],
        }

    elif device_type == "sts":
        by += ["sts_type"]
        additional_functions = {"sts_type": ["count"]}

    data_agg = aggregate_periodic_data(
        data=device_dataset,
        time_header="start_timestamp",
        window_start_header="start_window",
        window_stop_header="stop_window",
        categorical_columns=categorical_columns,
        continuous_columns=continuous_columns,
        period=period,
        freq=freq,
        thresh=thresh,
        by=by,
        statistics=statistics,
        additional_functions=additional_functions,
    )

    index = [col for col in data_agg.columns 
         if not col.startswith(device_type)]

    window_seconds = pd.to_timedelta(to_offset(period)).total_seconds()
    time_header = ["start_window", "stop_window"]

    if device_type == "sts":
        data_agg = data_agg.pivot_table(
            index=index, columns="sts_type", aggfunc="first"
        ).reset_index()

        data_agg.columns = [
            (
                f"sts_{sts_type}_{val.replace('sts_', '').replace('type_count', 'Events_count')}"
                if sts_type
                else val
            )
            for val, sts_type in data_agg.columns
        ]

        data_agg.sts_SitStand_Events_count = data_agg.sts_SitStand_Events_count.fillna(
            0
        )
        data_agg.sts_StandSit_Events_count = data_agg.sts_StandSit_Events_count.fillna(
            0
        )

    elif device_type == "mcl":

        if "mcl_status" in data_agg.columns:
            data_agg["mcl_class"] = (
                data_agg["mcl_status"].astype(str)
                + "_"
                + data_agg["mcl_class"].astype(str)
            )
            data_agg.drop(columns="mcl_status", inplace=True)

        data_agg.rename(
            columns={
                "mcl_duration_sum": "mcl_duration",
                "mcl_movement_time_sum": "mcl_movement_time",
            },
            inplace=True,
        )
        data_agg["mcl_total_duration"] = data_agg.groupby(["subject_id"] + time_header)[
            "mcl_duration"
        ].transform("sum")
        threshold_value = thresh * window_seconds
        data_agg = data_agg[data_agg.mcl_total_duration >= threshold_value]
        data_agg.drop(columns=["mcl_total_duration"], inplace=True)

        data_agg = data_agg.pivot_table(
            index=index, columns="mcl_class", aggfunc="first"
        ).reset_index()

        data_agg.columns = [
            (
                f"mcl_{mcl_class}_{val.replace('mcl_', '').replace('class_count', 'Events_count')}"
                if mcl_class
                else val
            )
            for val, mcl_class in data_agg.columns
        ]

        to_drop = [
            col
            for col in data_agg.columns
            if col.endswith("Events_count") and not col.endswith("shift_Events_count")
        ]
        data_agg.drop(
            columns=[
                col
                for col in data_agg.columns
                if col.endswith("shift_movement_time") or col in to_drop
            ],
            inplace=True,
        )
        data_agg = data_agg.fillna(0)

        # Add class mode
        duration_columns = [c for c in data_agg.columns if c.endswith("_duration")]
        position = (
            data_agg[duration_columns]
            .idxmax(axis=1)
            .str.replace("_duration", "")
            .str.replace("mcl_", "")
        )
        data_agg.insert(len(index), "mcl_class", position)

        # reorganise
        transition_cols = [
            c for c in data_agg.columns if c.startswith("mcl_transition")
        ]
        data_agg = data_agg[
            [c for c in data_agg.columns if c not in transition_cols] + transition_cols
        ]

    return data_agg.copy()
