def build_analysis_dataset(
    device_types,
    concept,
    window_size,
    change_from_baseline=False,
    statistics=None,
    window_length=2,
    setting="around",
    thresh=0.75,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    cohorts=None,
    covariates=True,
    savepath=None,
):
    """
    Build the analysis dataset by combining and aggregating device and diary datasets,
    incorporating the covariates specified in the SAP (demographics, longitudinal covariates,
    baselines of the PROs, and disease score if considering a single cohort),
    with the option to filter by cohort.

    :param device_types: A list of device types to analyze.
        Each must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR',
        'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_types: list of str

    :param concept: Must be one of 'F' (fatigue), 'SQ' (sleep quality), 'DS' (daytime sleepiness),
        or 'all' (to load all concepts).
    :type concept: str

    :param window_size: Specifies the timing of the questions we want to load.
        Must be one of 'intra_daily' or 'daily'.
    :type window_size: str

    :param change_from_baseline: If True, computes changes from baselines. Default is False.
    :type change_from_baseline: bool, optional

    :param statistics: Statistics for aggregation. Default is None, which uses ["mean", "std", "var", "max", "min", "25p", "50p", "75p", "mode"].
    :type statistics: list of str, optional

    :param window_length: Length of the aggregation window in hours. Only useful if window_size is 'intra_day'. Default is 2.
    :type window_length: int, optional

    :param setting: Must be one of 'before', 'around', 'after'. Default is 'around' (window_length/2 hours before, window_length/2 hours after).
        Only useful if window_size is 'intra_day'.
    :type setting: str, optional

    :param thresh: Minimum proportion of non-missing values required in a group for aggregation. Default is 0.75 (in range [0, 1]).
    :type thresh: float, optional

    :param version: Specifies the version of the data to use.
    :type version: str, optional

    :param imputation: Specifies whether the dataset to read should be imputed. Default is False.
    :type imputation: bool, optional

    :param analysis_set: Specifies whether to use 'ias' or 'eas'. Default is 'ias'.
    :type analysis_set: str, optional

    :param participants: List of participants to consider. Default is None, in which case all participants from the given dataset (ias or eas) are used.
    :type participants: str or list, optional

    :param ias_dataset: If participants is None, specifies the dataset to load ('Train', 'Test', or 'all'). Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param cohorts: List of cohorts to consider. Default is None, meaning all cohorts are included.
    :type cohorts: list of str, optional

    :param covariates: Add covariates to the dataframe. Default is True.
    :type covariates: bool, optional

    :param savepath: Local path to save the analysis dataset. Default is None, in which case the dataset is not saved.
    :type savepath: str, optional

    :return: The constructed analysis dataset.
    :rtype: pandas.DataFrame

    :note:
        - Combines device data and diary data based on timings and performs aggregation as specified by the parameters.
        - Covariates such as demographics, PROs, and disease scores are incorporated as specified in the SAP.
        - The function supports intra-day and daily window sizes and allows for the calculation of changes from baseline if requested.
        - The dataset is merged with diaries and saved (if specified) before being returned.

    :raises ValueError: If the cohorts or device_type are invalid.
    """

    cohorts = check_cohorts(cohorts)
    device_types = check_valid_devices(device_types)

    diaries = load_diaries(
        concept,
        window_size,
        change_from_baseline,
        version,
        imputation,
        analysis_set,
        participants,
        ias_dataset,
        cohorts,
    )

    timings = get_timings(diaries=diaries)

    if participants is not None:
        ias_dataset = "all"

    analysis_dataset = build_dataset_on_timings(
        device_types,
        timings,
        window_length,
        setting,
        statistics,
        thresh,
        version,
        imputation,
        analysis_set,
        covariates=covariates
    )

    diaries["Date"] = pd.to_datetime(diaries["Time"]).dt.date
    analysis_dataset["Date"] = pd.to_datetime(
        analysis_dataset.start_window).dt.date

    analysis_dataset = diaries.merge(analysis_dataset)

    analysis_dataset.drop(columns=["Date", "Time"], inplace=True)

    if savepath is not None:
        analysis_dataset.to_csv(savepath, index=False)

    return analysis_dataset.copy()

def build_dataset_on_timings(
    device_types,
    timings,
    window_length=2,
    setting="around",
    statistics=None,
    thresh=0.75,
    version=None,
    imputation=False,
    analysis_set="ias",
    covariates=True,
    savepath=None
):
    """
    Build the analysis dataset by aggregating device data based on timings, with the option to filter by cohort.

    :param device_types: A list of device types to analyze. Each must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR', 'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_types: list of str

    :param timings: A DataFrame that must contain at least the following columns: ["subject_id", "Time", "tup", "tup_day", "TIMING"].
    :type timings: pandas.DataFrame

    :param window_length: Length of the aggregation window in hours. Only useful if `setting` is 'intra_day'. Default is 2.
    :type window_length: int, optional

    :param setting: Defines the timing window for intra-day aggregation. Must be one of 'before', 'around', or 'after'. Default is 'around' (window_length/2 hours before, window_length/2 hours after).
    :type setting: str, optional

    :param statistics: The statistics to compute during aggregation. Default is None, in which case the aggregation is done with ["mean", "std", "var", "max", "min", "25p", "50p", "75p", "mode"].
    :type statistics: list of str, optional

    :param thresh: Minimum proportion of non-missing values required for aggregation. Value between 0 and 1. Default is 0.75.
    :type thresh: float, optional

    :param version: Specifies the version of the data to use.
    :type version: str, optional

    :param imputation: Specifies if the dataset to read must be imputed. Default is False.
    :type imputation: bool, optional

    :return: The analysis dataset with device data aggregated based on the specified timings.
    :rtype: pandas.DataFrame

    :param analysis_set: Specifies whether to use 'ias' or 'eas'. Default is 'ias'.
    :type analysis_set: str, optional

    :param covariates: Add covariates to the dataframe. Default is True.
    :type covariates: bool, optional

    :param savepath: Local path to save the analysis dataset. Default is None, in which case the dataset is not saved.
    :type savepath: str, optional

    :note:
        - The function handles different window settings ('before', 'around', 'after') for intra-day aggregation.
        - Device data is aggregated based on timings and merged with the `timings` DataFrame.
        - Covariates (e.g., demographics, baseline PROs) are added to the final dataset by the `add_covariates` function.
        - The `window_length` parameter controls the length of the aggregation window for intra-day data.

    :raises ValueError: If the `timings` DataFrame is missing required columns, or if an invalid `setting` is provided.
    """
    timings = timings.copy()

    if len(
        np.setdiff1d(
            ["subject_id", "tup", "tup_day", "TIMING", "Time"], timings.columns
        )
    ):
        raise ValueError(
            "DataFrame timings must contain columns {'subject_id', 'tup', 'tup_day', 'TIMING', 'Time'}."
        )

    window_size = "daily" if timings.TIMING.unique()[
        0] == "daily" else "intra_daily"

    timings.Time = pd.to_datetime(timings["Time"])

    if "Date" not in timings.columns:
        timings.insert(3, "Date", timings.Time.dt.date)

    timings = add_start_stop_window(timings,
                                    window_size,
                                    window_length,
                                    setting)

    participants = timings.subject_id.unique()
    for i, device_type in enumerate(device_types):
        device_dataset = read_device_dataset(
            device_type,
            version,
            imputation,
            analysis_set,
            participants,
        )
        device_dataset.drop(columns=["device_id"], inplace=True)

        if device_type in ["MCR_MCR", "MCR_MCL", "HRV", "VIT", "STS"]:
            if window_size == "intra_daily":
                dataset = aggregate_device_data_on_timings(
                    device_dataset,
                    timings,
                    window_length,
                    setting,
                    statistics,
                    thresh,
                )
            else:
                dataset = aggregate_device_data_on_frequency(
                    device_dataset,
                    "D",
                    statistics,
                    thresh,
                )
                dataset["Date"] = dataset["start_window"].dt.date

            dataset = timings.merge(dataset, how="inner")

        else:
            device_dataset["Date"] = pd.to_datetime(
                device_dataset["end_timestamp"]
            ).dt.date

            # Adjust 'tup_day' to reflect the day of awakening, to merge properly with diaries
            if device_type == "SLP_night":
                device_dataset.tup_day = device_dataset.tup_day + 1

            dataset = timings.merge(device_dataset, how="inner")
            dataset.drop(columns=["start_timestamp",
                         "end_timestamp"], inplace=True)

        dataset.drop(
            columns=[col for col in ["Date", "Time"]
                     if col in dataset.columns],
            inplace=True,
        )
        if i == 0:
            analysis_dataset = dataset
        else:
            analysis_dataset = analysis_dataset.merge(dataset, how="outer")

    if covariates:
        analysis_dataset = add_covariates(analysis_dataset,
                                          version,
                                          imputation,
                                          analysis_set,
                                          )
    if savepath is not None:
        analysis_dataset.to_csv(savepath, index=False)

    return analysis_dataset.copy()

def build_dataset_on_frequency(
    device_types,
    period="D",
    statistics=None,
    cohorts=None,
    thresh=0.75,
    version=None,
    imputation=False,
    analysis_set="ias",
    participants=None,
    ias_dataset="Train",
    covariates=True,
    savepath=None,
):
    """
    Build the unsupervised dataset by aggregating device data based on frequency, incorporating specified covariates (e.g., demographics, longitudinal covariates, baseline PROs, and disease score for a single cohort), with the option to filter by cohort.

    :param device_types: A list of device types to analyze. Each must be one of 'SLP_night', 'STS', 'HRV', 'VIT', 'GVA', 'MCR_MCR', 'MCR_MCL', 'COG_dst', 'COG_pvt'.
    :type device_types: list of str

    :param period: The frequency of aggregation. Default is "D" (daily).
    :type period: str, optional

    :param statistics: The statistics to compute during aggregation. Default is None, in which case the aggregation is done with ["mean", "std", "var", "max", "min", "25p", "50p", "75p", "mode"].
    :type statistics: list of str, optional

    :param cohorts: The list of cohorts to consider. Default is None, which means all cohorts are considered.
    :type cohorts: list of str, optional

    :param thresh: The minimum proportion of non-missing values required for aggregation. Value between 0 and 1. Default is 0.75.
    :type thresh: float, optional

    :param version: Specifies the version of the data to use.
    :type version: str, optional

    :param imputation: Specifies if the dataset to read must be imputed. Default is False.
    :type imputation: bool, optional

    :param analysis_set: Specifies whether to use 'ias' or 'eas'. Default is 'ias'.
    :type analysis_set: str, optional

    :param participants: List of participants to consider. Default is None, in which case all participants from the given dataset (ias or eas) are used.
    :type participants: str or list, optional

    :param ias_dataset: If participants is None, specifies the dataset to load ('Train', 'Test', or 'all'). Only useful if analysis_set="ias". Default is 'Train'.
    :type ias_dataset: str, optional

    :param covariates: Add covariates to the dataframe. Default is True.
    :type covariates: bool, optional

    :param savepath: Local path to save the analysis dataset to. If None, the dataset is not saved. Default is None.
    :type savepath: str, optional

    :return: The analysis dataset with device data aggregated on frequency, including covariates.
    :rtype: pandas.DataFrame

    :note:
        - The function handles different periods (e.g., daily or other frequencies) and adjusts the aggregation accordingly.
        - If a device type requires daily aggregation or other specific adjustments, those are handled within the function.
        - Covariates (e.g., demographics, baseline PROs) are added to the final dataset by the `add_covariates` function.
        - The final dataset can be saved to a local path if specified in the `savepath` argument.

    :raises ValueError: If invalid device types, cohorts or periods are provided.
    """

    cohorts = check_cohorts(cohorts)
    device_types = check_valid_devices(device_types)
    period = check_period(period)
    daily = is_daily_period(period)
    day_factor = daily and period != "1D"

    sorted_device_types = [
        "MCR_MCR", "MCR_MCL", "HRV", "VIT", "STS",
        "SLP_night", "GVA", "COG_dst", "COG_pvt"
    ]
    device_types = sorted(
        device_types, key=lambda x: sorted_device_types.index(x))

    for i, device_type in enumerate(device_types):
        device_dataset = read_device_dataset(
            device_type,
            version=version,
            imputation=imputation,
            analysis_set=analysis_set,
            participants=participants,
            ias_dataset=ias_dataset,
            cohorts=cohorts,
        )
        device_dataset.drop(columns=["device_id"], inplace=True)

        # Adjust 'tup_day' to reflect the day of awakening
        if device_type == "SLP_night":
            device_dataset.tup_day = device_dataset.tup_day + 1

        if day_factor or device_type in ["MCR_MCR", "MCR_MCL", "HRV", "VIT", "STS"]:
            dataset = aggregate_device_data_on_frequency(
                device_dataset,
                period,
                statistics,
                thresh,
            )
            dataset.insert(3, "Date", pd.to_datetime(
                dataset.start_window).dt.date)
        else:
            device_dataset.insert(
                3, "Date", pd.to_datetime(device_dataset.end_timestamp).dt.date
            )
            dataset = device_dataset.drop(
                columns=["start_timestamp", "end_timestamp"])

        if i == 0:
            analysis_dataset = dataset
        else:
            analysis_dataset = analysis_dataset.merge(dataset, how="outer")

    if "start_window" not in analysis_dataset.columns:  # in case there is no aggregated data
        analysis_dataset.insert(3,
                                "start_window",
                                pd.NaT)
        analysis_dataset.insert(4,
                                "stop_window",
                                pd.NaT)

    analysis_dataset["start_window"] = analysis_dataset["start_window"].fillna(
        pd.to_datetime(analysis_dataset["Date"]).dt.floor("D")
    )

    analysis_dataset["stop_window"] = analysis_dataset["stop_window"].fillna(
        pd.to_datetime(analysis_dataset["Date"]).dt.floor(
            "D") + pd.Timedelta(days=1)
    )

    analysis_dataset.drop(columns="Date", inplace=True)

    if analysis_dataset.empty:
        return None

    if covariates:
        analysis_dataset = add_covariates(analysis_dataset,
                                          version,
                                          imputation,
                                          analysis_set,
                                          )

    if savepath is not None:
        analysis_dataset.to_csv(savepath, index=False)

    return analysis_dataset.copy()

def add_covariates(analysis_dataset,
                   version,
                   imputation,
                   analysis_set,
                   ):
    """
    Add covariates to the analysis dataset by merging relevant demographic and clinical data,
    longitudinal covariates, and baseline PROs.

    :param analysis_dataset: The dataset to which covariates will be added.
                             It should contain at least the 'subject_id', 'tup' and 'start_window' columns.
    :type analysis_dataset: pandas.DataFrame

    :param version: Specifies the version of the data to use.
    :type version: str

    :param analysis_set: "ias" or "eas". Default is "ias".
    :type analysis_set: str, optional

    :return: The modified dataset with added covariates, including demographics,
             longitudinal covariates, baseline PROs, and disease scores (if applicable).
    :rtype: pandas.DataFrame

    :note:
        - Demographics, longitudinal covariates, and baseline PROs are merged using the 'subject_id'
          and 'Date' columns.
        - Disease scores are merged based on 'subject_id' and 'tup' if a single cohort is provided.
          If multiple cohorts are considered, the disease score and TIMESINCEDIAG information is excluded.
    """

    analysis_dataset = analysis_dataset.copy()

    analysis_dataset["Date"] = pd.to_datetime(
        analysis_dataset.start_window).dt.date

    participants = analysis_dataset.subject_id.unique().tolist()

    # Add demographics
    demographics = get_demographics(
        version, imputation, analysis_set, participants)
    analysis_dataset = analysis_dataset.merge(
        demographics, on="subject_id", how="left")

    # Add covariates
    longitudinal_covariates = get_longitudinal_covariates(analysis_dataset)
    analysis_dataset = analysis_dataset.merge(
        longitudinal_covariates, on=["subject_id", "Date"], how="left"
    )

    baseline_PROs = get_baseline_PROs(
        version, imputation, analysis_set, participants)
    analysis_dataset = analysis_dataset.merge(
        baseline_PROs, on=["subject_id"], how="left"
    )

    cohorts = analysis_dataset.COHORT.unique().tolist()
    if len(cohorts) == 1:
        disease_score = get_disease_score(version=version,
                                          imputation=imputation,
                                          analysis_set=analysis_set,
                                          participants=participants,
                                          cohorts=cohorts)
        disease_score = disease_score.pivot(
            index=["subject_id", "tup"], columns="Disease_test", values="score"
        ).reset_index()
        disease_score.columns.name = None
        analysis_dataset = analysis_dataset.merge(
            disease_score, on=["subject_id", "tup"], how="left"
        )
    else:
        analysis_dataset.drop(columns=["TIMESINCEDIAG"], inplace=True)

    analysis_dataset.drop(columns="Date", inplace=True)

    return analysis_dataset.copy()

def add_start_stop_window(timings, window_size, window_length, setting):
    """
    Add the proper start and stop windows corresponding to the timings.

    :param timings: A DataFrame that must contain at least the following columns: ["subject_id", "Time", "tup", "tup_day", "TIMING"].
    :type timings: pandas.DataFrame

    :param window_size: Specifies the timing of the questions we want to load.
        Must be one of 'intra_daily' or 'daily'.
    :type window_size: str

    :param window_length: Length of the aggregation window in hours. Only useful if window_size is 'intra_day'. Default is 2.
    :type window_length: int

    :param setting: Must be one of 'before', 'around', 'after'. Default is 'around' (window_length/2 hours before, window_length/2 hours after).
        Only useful if window_size is 'intra_day'.
    :type setting: str
    """
    if window_size == "intra_daily":
        start, end = calculate_start_end(timings, window_length, setting)
    else:
        start = pd.to_datetime(timings.Date).dt.floor("D")
        end = start + pd.Timedelta(days=1)

    timings.insert(3, "start_window", start)
    timings.insert(4, "stop_window", end)

    return timings
